 18 function LoadingPanel(message) {
 19 	// TODO: make loading panel module
 20     this.panel = new YAHOO.widget.Panel("wait", {
 21         width: "240px",
 22         fixedcenter: true,
 23         close: false,
 24         draggable: false,
 25         visible: false,
 26         constraintoviewport: true,
 27         iframe: true
 28     });
 29 
 30     this.panel.setHeader(message);
 31     this.panel.setBody("<img src='images/loading.gif'>");
 32     this.panel.render(document.getElementById("mainBody"));
 33 }
 34 LoadingPanel.prototype.show = function () {
 35     this.panel.show();
 36 };
 37 LoadingPanel.prototype.hide = function () {
 38     this.panel.hide();
 39 };
