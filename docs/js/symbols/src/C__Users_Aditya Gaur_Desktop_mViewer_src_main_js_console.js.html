 17 YUI({
 18 		filter : 'raw'
 19 	}).use("console", "console-filters", "dd-plugin", function(Y) {
 20 		// creating a console screen
 21 		var globalConsole;
 22 		globalConsole = new Y.Console({
 23 			logSource : Y.Global,
 24 			strings : {
 25 				title : 'Console',
 26 				pause : 'Pause',
 27 				clear : 'Clear',
 28 				collapse : 'Collapse',
 29 				expand : 'Expand'
 30 			},
 31 			visible : false
 32 		}).plug(Y.Plugin.ConsoleFilters).plug(Y.Plugin.Drag, {
 33 			handles : ['.yui3-console-hd']
 34 		}).render();
 35 		// Set up the button listeners
 36 		function toggle(e, globalConsole) {
 37 			if (globalConsole.get('visible')) {
 38 				globalConsole.hide();
 39 				this.set('innerHTML', 'Show console');
 40 			} else {
 41 				globalConsole.show();
 42 				globalConsole.syncUI(); // to handle any UI changes queued while hidden.
 43 				this.set('innerHTML', 'Hide console');
 44 			}
 45 		}
 46 		Y.on('click', toggle, '#toggle_console', null, globalConsole);
 47 	});
