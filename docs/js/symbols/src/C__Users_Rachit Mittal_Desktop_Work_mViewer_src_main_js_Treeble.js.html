 12     DS.TYPE_TREELIST = 9;
 13 /*
 14 
 15     Each element in this._open contains information about an openable,
 16     top-level node and is the root of a tree of open (or previously opened)
 17     items.  Each node in a tree contains the following data:
 18 
 19         index:      {Number} sorting key; the index of the node
 20         open:       null if never opened, true if open, false otherwise
 21         ds:         {DataSource} source for child nodes
 22         childTotal: {Number} total number of child nodes
 23         children:   {Array} (recursive) child nodes which are or have been opened
 24         parent:     {Object} parent item
 25 
 26     Each level is sorted by index to allow simple traversal in display
 27     order.
 28 
 29  */

 31     /**
 32      * <p>TreebleDataSource converts a tree of DataSources into a flat list of
 33      * visible items.  This list of items can then be paginated by DataTable.
 34      * The merged list must be paginated if the number of child nodes might be
 35      * very large.  To turn on this feature, set paginateChildren:true.</p>
 36      *
 37      * <p>The tree must be immutable.  The total number of items available from
 38      * each DataSource must remain constant.</p>
 39      *
 40      * @module Treeble
 41      * @namespace YAHOO.util
 42      * @class YAHOO.util.TreebleDataSource
 43      * @extends YAHOO.util.DataSourceBase
 44      * @constructor
 45      * @param oLiveData {DataSource}  The top-level DataSource.
 46      *        You must pass a treebleConfig object as part of this object's configuration.
 47      *        This object can contain the following configuration:
 48      *        <dl>
 49      *        <dt>generateRequest</dt>
 50      *        <dd>(required) The function to convert the output from
 51      *            <code>DataTable.generateTreebleDataSourceRequest()</code> into
 52      *            a request usable by one of the actual DataSources.  This function
 53      *            takes two arguments: state (sort,dir,startIndex,results) and path
 54      *            (an array of node indices telling how to reach the node).
 55      *            </dd>
 56      *        <dt>childNodesKey</dt>
 57      *        <dd>(semi-optional) The name of the key inside a node which contains
 58      *            the data used to construct the DataSource for retrieving the children.
 59      *            This config is only required if you provide a custom parser.</dd>
 60      *        <dt>startIndexExpr</dt>
 61      *        <dd>(optional) OGNL expression telling how to extract the startIndex
 62      *            from the received data, e.g., <code>.meta.startIndex</code>.
 63      *            If it is not provided, startIndex is always assumed to be zero.</dd>
 64      *        <dt>totalRecordsExpr</dt>
 65      *        <dd>(semi-optional) OGNL expression telling how to extract the total number
 66      *            of records from the received data, e.g., <code>.meta.totalRecords</code>.
 67      *            If this is not provided, <code>totalRecordsReturnExpr</code> must be
 68      *            specified.</dd>
 69      *        <dt>totalRecordsReturnExpr</dt>
 70      *        <dd>(semi-optional) OGNL expression telling where in the response to store
 71      *            the total number of records, e.g., <code>.meta.totalRecords</code>.
 72      *            This is only appropriate for DataSources that always return the
 73      *            entire data set.  If this is not provided,
 74      *            <code>totalRecordsExpr</code> must be specified.  If both are provided,
 75      *            <code>totalRecordsExpr</code> takes priority.</dd>
 76      *        </dl>
 77      * @param oConfigs {Object} Object literal of configuration values.
 78      *        <dl>
 79      *        <dt>paginateChildren</dt>
 80      *        <dd>(optional) Pass <code>true</code> to paginate the result after merging
 81      *            child nodes into the list.  The default (<code>false</code>) is to
 82      *            paginate only top-level nodes, so all children are visible.</dd>
 83      *        <dt>uniqueIdKey</dt>
 84      *        <dd>(optional) The key in each record that stores an identifier which is
 85      *            unique across the entire tree.  If this is not specified, then
 86      *            all nodes will close when the data is sorted.</dd>
 87      *        </dl>
 88      */
 89     util.TreebleDataSource = function(oLiveData, oConfigs) {
 90         if (!oLiveData instanceof util.DataSourceBase) {
 91             YAHOO.log('TreebleDataSource requires DataSource', 'error', 'TreebleDataSource');
 92             return;
 93         }
 94 
 95         if (!oLiveData.treebleConfig.childNodesKey) {
 96             var fields = oLiveData.responseSchema.fields,
 97                 i;
 98             for (i = 0; i < fields.length; i++) {
 99                 if (lang.isObject(fields[i]) && fields[i].parser === 'datasource') {
100                     oLiveData.treebleConfig.childNodesKey = fields[i].key;
101                     break;
102                 }
103             }
104 
105             if (!oLiveData.treebleConfig.childNodesKey) {
106                 YAHOO.log('TreebleDataSource requires treebleConfig.childNodesKey configuration to be set on top-level DataSource', 'error', 'TreebleDataSource');
107                 return;
108             }
109         }
110 
111         if (!oLiveData.treebleConfig.generateRequest) {
112             YAHOO.log('TreebleDataSource requires treebleConfig.generateRequest configuration to be set on top-level DataSource', 'error', 'TreebleDataSource');
113             return;
114         }
115 
116         if (!oLiveData.treebleConfig.totalRecordsExpr && !oLiveData.treebleConfig.totalRecordsReturnExpr) {
117             YAHOO.log('TreebleDataSource requires either treebleConfig.totalRecordsExpr or treebleConfig.totalRecordsReturnExpr configuration to be set on top-level DataSource', 'error', 'TreebleDataSource');
118             return;
119         }
120 
121         this.dataType = DS.TYPE_TREELIST;
122         this._open = [];
123         this._open_cache = {};
124         this._req = [];
125         util.TreebleDataSource.superclass.constructor.call(this, oLiveData, oConfigs);
126     };

128     function populateOpen( /* object */ parent, /* array */ open, /* array */ data, /* int */ startIndex, /* string */ childNodesKey) {
129         var j;
130         for (j = 0; j < open.length; j++) {
131             if (open[j].index >= startIndex) {
132                 break;
133             }
134         }
135 
136         var result = true,
137             k;
138         for (k = 0; k < data.length; k++) {
139             var i = startIndex + k;
140             var ds = data[k][childNodesKey];
141             if (!ds) {
142                 continue;
143             }
144 
145             while (j < open.length && open[j].index < i) {
146                 open.splice(j, 1);
147                 result = false;
148 
149                 if (this.uniqueIdKey) {
150                     delete this._open_cache[data[k][this.uniqueIdKey]];
151                 }
152             }
153 
154             if (j >= open.length || open[j].index > i) {
155                 var item = {
156                     index: i,
157                     open: null,
158                     ds: ds,
159                     children: [],
160                     childTotal: 0,
161                     parent: parent
162                 };
163 
164                 if (this.uniqueIdKey) {
165                     var cached_item = this._open_cache[data[k][this.uniqueIdKey]];
166                     if (cached_item) {
167                         item.open = cached_item.open;
168                         item.childTotal = cached_item.childTotal;
169                         this._redo = this._redo || item.open;
170                     }
171                 }
172 
173                 open.splice(j, 0, item);
174                 this._open_cache[data[k][this.uniqueIdKey]] = item;
175             }
176 
177             j++;
178         }
179 
180         return result;
181     }

185     function searchOpen( /* array */ list, /* int */ nodeIndex) {
186         var i;
187         for (i = 0; i < list.length; i++) {
188             if (list[i].index === nodeIndex) {
189                 return list[i];
190             }
191         }
192 
193         return false;

196     function getNode( /* array */ path) {
197         var open = this._open;
198         var last = path.length - 1;
199         var i;
200         for (i = 0; i < last; i++) {
201             var node = searchOpen(open, path[i]);
202             open = node.children;
203         }
204 
205         return searchOpen(open, path[last]);
206     }

208     function countVisibleNodes(
209 
210     // not sent by initiator
211     /* array */ open) {
212         var total = 0;
213         if (!open) {
214             open = this._open;
215             total = this._topNodeTotal;
216         }
217 
218         if (this.paginateChildren) {
219             var i;
220             for (i = 0; i < open.length; i++) {
221                 var node = open[i];
222                 if (node.open) {
223                     total += node.childTotal;
224                     total += countVisibleNodes.call(this, node.children);
225                 }
226             }
227         }
228 
229         return total;
230     }

232     function requestTree() {
233         this._cancelAllRequests();
234 
235         this._redo = false;
236         this._generating_requests = true;
237 
238         var req = this._callback.request;
239         if (this.paginateChildren) {
240             this._slices = getVisibleSlicesPgAll(req.startIndex, req.results, this.liveData, this._open);
241         } else {
242             this._slices = getVisibleSlicesPgTop(req.startIndex, req.results, this.liveData, this._open);
243         }
244 
245         requestSlices.call(this, req);
246 
247         this._generating_requests = false;
248         checkFinished.call(this);
249     }

251     function getVisibleSlicesPgTop( /* int */ skip, /* int */ show, /* DataSource */ ds, /* array */ open,
252 
253     // not sent by initiator
254     /* array */ path) {
255         open = open.concat({
256             index: -1,
257             open: true,
258             childTotal: 0,
259             children: null
260         });
261 
262         if (!path) {
263             path = [];
264         }
265 
266         var slices = [],
267             send = false;
268 
269         var m = 0,
270             prev = -1,
271             presend = false,
272             i;
273         for (i = 0; i < open.length; i++) {
274             var node = open[i];
275             if (!node.open) {
276                 continue;
277             }
278 
279             var delta = node.index - prev;
280 
281             if (m + delta >= skip + show || node.index === -1) {
282                 slices.push({
283                     ds: ds,
284                     path: path.slice(0),
285                     start: send ? m : skip,
286                     end: skip + show - 1
287                 });
288 
289                 if (m + delta === skip + show) {
290                     slices = slices.concat(
291                     getVisibleSlicesPgTop(0, node.childTotal, node.ds, node.children, path.concat(node.index)));
292                 }
293 
294                 return slices;
295             } else if (!send && m + delta === skip) {
296                 presend = true;
297             } else if (m + delta > skip) {
298                 slices.push({
299                     ds: ds,
300                     path: path.slice(0),
301                     start: send ? prev + 1 : skip,
302                     end: m + delta - 1
303                 });
304                 send = true;
305             }
306 
307             m += delta;
308 
309             if (send && node.childTotal > 0) {
310                 slices = slices.concat(
311                 getVisibleSlicesPgTop(0, node.childTotal, node.ds, node.children, path.concat(node.index)));
312             }
313 
314             prev = node.index;
315             send = send || presend;
316         }
317     }

319     function getVisibleSlicesPgAll( /* int */ skip, /* int */ show, /* DataSource */ rootDS, /* array */ open,
320 
321     // not sent by initiator
322     /* array */ path, /* node */ parent, /* int */ pre, /* bool */ send, /* array */ slices) {
323         if (!parent) {
324             path = [];
325             parent = null;
326             pre = 0;
327             send = false;
328             slices = [];
329         }
330 
331         var ds = parent ? parent.ds : rootDS;
332 
333         open = open.concat({
334             index: parent ? parent.childTotal : -1,
335             open: true,
336             childTotal: 0,
337             children: null
338         });
339 
340         var n = 0,
341             m = 0,
342             prev = -1,
343             i;
344         for (i = 0; i < open.length; i++) {
345             var node = open[i];
346             if (!node.open) {
347                 continue;
348             }
349 
350             var delta = node.index - prev;
351             if (node.children === null) {
352                 delta--; // last item is off the end
353             }
354 
355             if (pre + n + delta >= skip + show || node.index === -1) {
356                 slices.push({
357                     ds: ds,
358                     path: path.slice(0),
359                     start: m + (send ? 0 : skip - pre - n),
360                     end: m + (skip + show - 1 - pre - n)
361                 });
362 
363                 return slices;
364             } else if (!send && pre + n + delta === skip) {
365                 send = true;
366             } else if (pre + n + delta > skip) {
367                 slices.push({
368                     ds: ds,
369                     path: path.slice(0),
370                     start: m + (send ? 0 : skip - pre - n),
371                     end: m + delta - 1
372                 });
373                 send = true;
374             }
375 
376             n += delta;
377             m += delta;
378 
379             if (node.childTotal > 0) {
380                 var info = getVisibleSlicesPgAll(skip, show, rootDS, node.children, path.concat(node.index), node, pre + n, send, slices);
381                 if (info instanceof Array) {
382                     return info;
383                 } else {
384                     n += info.count;
385                     send = info.send;
386                 }
387             }
388 
389             prev = node.index;
390         }
391 
392         // only reached when parent != null
393         var information = {
394             count: n,
395             send: send
396         };
397         return information;
398     }

400     function requestSlices( /* object */ request) {
401         var i, req;
402         for (i = 0; i < this._slices.length; i++) {
403             var slice = this._slices[i];
404             var ds = slice.ds;
405             req = findRequest.call(this, ds);
406             if (req) {
407                 if (YAHOO.widget.Logger) {
408                     if (req.end + 1 < slice.start) {
409                         YAHOO.log('TreebleDataSource found discontinuous range', 'error', 'TreebleDataSource');
410                     }
411 
412                     if (req.path.length !== slice.path.length) {
413                         YAHOO.log('TreebleDataSource found path length mismatch', 'error', 'TreebleDataSource');
414                     } else {
415                         var j;
416                         for (j = 0; j < slice.path.length; j++) {
417                             if (req.path[j] !== slice.path[j]) {
418                                 YAHOO.log('TreebleDataSource found path mismatch', 'error', 'TreebleDataSource');
419                                 break;
420                             }
421                         }
422                     }
423                 }
424 
425                 req.end = slice.end;
426             } else {
427                 this._req.push({
428                     ds: ds,
429                     path: slice.path,
430                     start: slice.start,
431                     end: slice.end
432                 });
433             }
434         }
435 
436         request = cloneObject(request);
437         for (i = 0; i < this._req.length; i++) {
438             req = this._req[i];
439             request.startIndex = req.start;
440             request.results = req.end - req.start + 1;
441 
442             req.txId = req.ds.sendRequest(req.ds.treebleConfig.generateRequest(request, req.path), {
443                 success: treeSuccess,
444                 failure: treeFailure,
445                 scope: this,
446                 argument: i
447             });
448         }
449     }

451     function findRequest( /* DataSource */ ds) {
452         var i;
453         for (i = 0; i < this._req.length; i++) {
454             var req = this._req[i];
455             if (ds === req.ds) {
456                 return req;
457             }
458         }
459 
460         return null;
461     }

463     function treeSuccess(oRequest, oParsedResponse, reqIndex) {
464         if (!oParsedResponse || oParsedResponse.error || !(oParsedResponse.results instanceof Array)) {
465             treeFailure.apply(this, arguments);
466             return;
467         }
468 
469         var req = searchTxId(this._req, oParsedResponse.tId, reqIndex);
470         if (!req) {
471             return; // cancelled request
472         }
473 
474         if (!this._topResponse && req.ds === this.liveData) {
475             this._topResponse = oParsedResponse;
476         }
477 
478         req.txId = null;
479         req.resp = oParsedResponse;
480         req.error = false;
481 
482         var dataStartIndex = 0;
483         if (req.ds.treebleConfig.startIndexExpr) {
484             eval('dataStartIndex=req.resp' + req.ds.treebleConfig.startIndexExpr);
485         }
486 
487         var sliceStartIndex = req.start - dataStartIndex;
488         req.data = oParsedResponse.results.slice(sliceStartIndex, req.end - dataStartIndex + 1);
489         setNodeInfo(req.data, req.start, req.path, req.ds);
490 
491         var parent = (req.path.length > 0 ? getNode.call(this, req.path) : null);
492         var open = (parent !== null ? parent.children : this._open);
493         if (!populateOpen.call(this, parent, open, req.data, req.start, req.ds.treebleConfig.childNodesKey)) {
494             treeFailure.apply(this, arguments);
495             return;
496         }
497 
498         if (!parent && req.ds.treebleConfig.totalRecordsExpr) {
499             eval('this._topNodeTotal=oParsedResponse' + req.ds.treebleConfig.totalRecordsExpr);
500         } else if (!parent && req.ds.treebleConfig.totalRecordsReturnExpr) {
501             this._topNodeTotal = oParsedResponse.results.length;
502         }
503 
504         checkFinished.call(this);
505     }
506 
507     function treeFailure(oRequest, oParsedResponse, reqIndex) {
508         var req = searchTxId(this._req, oParsedResponse.tId, reqIndex);
509         if (!req) {
510             return; // cancelled request
511         }
512 
513         this._cancelAllRequests();
514         DS.issueCallback(this._callback.callback, [this._callback.request, oParsedResponse], true, this._callback.caller);
515     }

522             list[i]._yui_node_path = path.concat(offset + i);
523             list[i]._yui_node_ds = ds;

527     function searchTxId( /* array */ req, /* int */ id, /* int */ fallbackIndex) {
528         var i;
529         for (i = 0; i < req.length; i++) {
530             if (req[i].txId === id) {
531                 return req[i];
532             }
533         }
534 
535         // synch response arrives before setting txId
536         if (fallbackIndex < req.length && lang.isUndefined(req[fallbackIndex].txId)) {
537             return req[fallbackIndex];
538         }
539 
540         return null;
541     }

543     function checkFinished() {
544         if (this._generating_requests) {
545             return;
546         }
547 
548         var count = this._req.length,
549             i;
550         for (i = 0; i < count; i++) {
551             if (!this._req[i].resp) {
552                 return;
553             }
554         }
555 
556         if (this._redo) {
557             YAHOO.lang.later(0, this, requestTree);
558             return;
559         }
560 
561         var response = {};
562         lang.augmentObject(response, this._topResponse);
563         response.results = [];
564         response = cloneObject(response);
565 
566         count = this._slices.length;
567         for (i = 0; i < count; i++) {
568             var slice = this._slices[i];
569             var req = findRequest.call(this, slice.ds);
570             if (!req) {
571                 YAHOO.log('Failed to find request for a slice', 'error', 'TreebleDataSource');
572                 continue;
573             }
574 
575             var j = slice.start - req.start;
576             var data = req.data.slice(j, j + slice.end - slice.start + 1);
577 
578             response.results = response.results.concat(data);
579         }
580 
581         if (this.liveData.treebleConfig.totalRecordsExpr) {
582             eval('response' + this.liveData.treebleConfig.totalRecordsExpr + '=' + countVisibleNodes.call(this));
583         } else if (this.liveData.treebleConfig.totalRecordsReturnExpr) {
584             eval('response' + this.liveData.treebleConfig.totalRecordsReturnExpr + '=' + countVisibleNodes.call(this));
585         }
586 
587         DS.issueCallback(this._callback.callback, [this._callback.request, response], false, this._callback.caller);
588     }

590     function cloneObject(o) {
591         if (!lang.isValue(o)) {
592             return o;
593         }
594 
595         var copy = {};
596 
597         if ((o instanceof RegExp) || lang.isFunction(o)) {
598             copy = o;
599         } else if (o instanceof Date) {
600             copy = new Date(o);
601         } else if (lang.isArray(o)) {
602             var array = [];
603             var i;
604             for (i = 0, len = o.length; i < len; i++) {
605                 array[i] = cloneObject(o[i]);
606             }
607             copy = array;
608         } else if (lang.isObject(o)) {
609             var x;
610             for (x in o) {
611                 if (lang.hasOwnProperty(o, x)) {
612                     if ((lang.isValue(o[x]) && lang.isObject(o[x])) || lang.isArray(o[x])) {
613                         copy[x] = cloneObject(o[x]);
614                     } else {
615                         copy[x] = o[x];
616                     }
617                 }
618             }
619         } else {
620             copy = o;
621         }
622 
623         return copy;
624     }

626     function toggleSuccess(oRequest, oParsedResponse, args) {
627         var node = args[0];
628         var completion = args[1];
629 
630         if (node.ds.treebleConfig.totalRecordsExpr) {
631             eval('node.childTotal=oParsedResponse' + node.ds.treebleConfig.totalRecordsExpr);
632         } else if (node.ds.treebleConfig.totalRecordsReturnExpr) {
633             node.childTotal = oParsedResponse.results.length;
634         }
635 
636         node.open = true;
637         node.children = [];
638         complete(completion);
639     }

641     function toggleFailure(oRequest, oParsedResponse, args) {
642         var node = args[0];
643         var completion = args[1];
644 
645         node.childTotal = 0;
646 
647         node.open = true;
648         node.children = [];
649         complete(completion);
650     }

652     function complete(f) {
653         if (YAHOO.lang.isFunction(f)) {
654             f();
655         } else if (f && f.fn) {
656             f.fn.apply(f.scope || window, f.args);
657         }

662         /**
663          * @param path {Array} Path to node
664          * @return {boolean} true if the node is open
665          */
666         isOpen: function(path) {
667             var list = this._open,
668                 i;
669             for (i = 0; i < path.length; i++) {
670                 var node = searchOpen.call(this, list, path[i]);
671                 if (!node || !node.open) {
672                     return false;
673                 }
674                 list = node.children;
675             }

680         /**
681          * Toggle the specified node between open and closed.  When a node is
682          * opened for the first time, this requires a request to the
683          * DataSource.  Any code that assumes the node has been opened must be
684          * passed in as a completion function.
685          *
686          * @param path {Array} Path to the node
687          * @param request {Object} Request generated by DT.generateTreebleDataSourceRequest()
688          * @param completion {Function|Object} Function to call when the operation completes.  Can be object: {fn,scope,args}
689          * @return {boolean} false if the path to the node has not yet been fully explored or is not openable, true otherwise
690          */
691         toggle: function(path, request, completion) {
692             var list = this._open,
693                 i, node;
694             for (i = 0; i < path.length; i++) {
695                 node = searchOpen.call(this, list, path[i]);
696                 if (!node) {
697                     return false;
698                 }
699                 list = node.children;
700             }
701 
702             if (node.open === null) {
703                 request.startIndex = 0;
704                 request.results = 0;
705                 node.ds.sendRequest(node.ds.treebleConfig.generateRequest(request, path), {
706                     success: toggleSuccess,
707                     failure: toggleFailure,
708                     scope: this,
709                     argument: [node, completion]
710                 });
711             } else {
712                 node.open = !node.open;
713                 complete(completion);
714             }
715             return true;
716         },

718         /**
719          * Overriding method generates queries to all visible DataSources.
720          *
721          * @method makeConnection
722          * @param oRequest {Object} Request object.
723          * @param oCallback {Object} Callback object literal.
724          * @param oCaller {Object} (deprecated) Use oCallback.scope.
725          * @return {Number} Transaction ID.
726          * @private
727          */
728         makeConnection: function(oRequest, oCallback, oCaller) {
729             var tId = DS._nTransactionId++;
730             this.fireEvent("requestEvent", {
731                 tId: tId,
732                 request: oRequest,
733                 callback: oCallback,
734                 caller: oCaller
735             });
736 
737             if (this._callback) {
738                 var r = this._callback.request;
739                 var key;
740                 for (key in r) {
741                     if (!YAHOO.lang.hasOwnProperty(r, key) || key === 'startIndex' || key === 'results') {
742                         continue;
743                     }
744 
745                     if (r[key] !== oRequest[key]) {
746                         this._open = [];
747                         break;
748                     }
749                 }
750             }
751 
752             this._callback = {
753                 request: oRequest,
754                 callback: oCallback,
755                 caller: oCaller
756             };
757 
758             requestTree.call(this);
759             return tId;
760         },

763             this._req = [];
764         }

770     /**
771      * Converts data to a DataSource.  Data can be an object containing both
772      * <code>dataType</code> and <code>liveData</code>, or it can be <q>free
773      * form</q>, e.g., an array of records or an XHR URL.
774      *
775      * @method DataSourceBase.parseDataSource
776      * @param oData {mixed} Data to convert.
777      * @return {DataSource} The new data source.
778      * @static
779      */
780     DS.parseDataSource = function(oData) {
781         var treebleConfig;
782         if (!oData) {
783             return null;
784         } else if (oData.dataType === DS.TYPE_JSFUNCTION) {
785             var fn = oData.liveData,
786                 scope;
787             if (lang.isString(fn)) {
788                 fn = window[fn];
789             } else if (fn.scope) {
790                 scope = fn.scope;
791                 fn = fn.fn;
792             }
793 
794             var ds = new util.FunctionDataSource(fn, {
795                 responseSchema: this.responseSchema,
796                 maxCacheEntries: this.maxCacheEntries,
797                 treebleConfig: this.treebleConfig
798             });
799 
800             if (scope) {
801                 ds.scope = scope;
802             }
803             return ds;
804         } else if (!lang.isUndefined(oData.dataType)) {
805             treebleConfig = this.treebleConfig;
806             if (oData.dataType === DS.TYPE_JSARRAY || oData.dataType === DS.TYPE_JSON) {
807                 treebleConfig = cloneObject(treebleConfig);
808                 delete treebleConfig.startIndexExpr;
809                 delete treebleConfig.totalRecordsExpr;
810             }
811 
812             return new util.DataSource(oData.liveData, {
813                 dataType: oData.dataType,
814                 responseSchema: this.responseSchema,
815                 maxCacheEntries: this.maxCacheEntries,
816                 treebleConfig: treebleConfig
817             });
818         } else {
819             treebleConfig = this.treebleConfig;
820             if (lang.isArray(oData)) {
821                 treebleConfig = cloneObject(treebleConfig);
822                 delete treebleConfig.startIndexExpr;
823                 delete treebleConfig.totalRecordsExpr;
824             }
825 
826             return new util.DataSource(oData, {
827                 responseSchema: this.responseSchema,
828                 maxCacheEntries: this.maxCacheEntries,
829                 treebleConfig: treebleConfig
830             });
831         }
832     };

864      * Toggle the state of a row.
865      *
866      * @method toggleRow
867      * @param path {Array} Path to node.
868      */
869     DT.prototype.toggleRow = function(path) {
870         if (!this._oDataSource instanceof util.TreebleDataSource) {
871             YAHOO.log('Treeble requires TreebleDataSource', 'error', 'Treeble');
872             return;
873         }
874 
875         // Get current state
876         var oState = this.getState();
877 
878         // Get the request for the new state
879         var oRequest = this.get("generateRequest")(oState, this);
880 
881         // Update state of treelist
882         this._oDataSource.toggle(path, oRequest, {
883             scope: this,
884             args: [oState, cloneObject(oRequest)],
885             fn: function(oState, oRequest) {
886                 // Purge selections
887                 this.unselectAllRows();
888                 this.unselectAllCells();
889 
890                 // Send request for new data
891                 var callback = {
892                     success: this.onDataReturnSetRows,
893                     failure: this.onDataReturnSetRows,
894                     argument: oState,
895                     // Pass along the new state to the callback
896                     scope: this
897                 };
898                 this._oDataSource.sendRequest(oRequest, callback);
899             }
900         });

907      * @param oState {Object} Pagination and sorting state.
908      * @param oSelf {DataTable} The data table.
909      * @return {Object} Pagination and sorting state formatted for TreebleDataSource.
910      * @static
911      */
912     DT.generateTreebleDataSourceRequest = function(oState, oSelf) {
913         // Set defaults
914         oState = oState || {
915             pagination: null,
916             sortedBy: null
917         };
918         var sort = encodeURIComponent((oState.sortedBy) ? oState.sortedBy.key : oSelf.getColumnSet().keys[0].getKey());
919         var dir = (oState.sortedBy && oState.sortedBy.dir === YAHOO.widget.DataTable.CLASS_DESC) ? "desc" : "asc";
920         var startIndex = (oState.pagination) ? oState.pagination.recordOffset : 0;
921         var results = (oState.pagination) ? oState.pagination.rowsPerPage : null;
922 
923         // Build the request
924         var state = {
925             sort: sort,
926             dir: dir,
927             startIndex: startIndex,
928             results: results
929         };
930         return state;
931     };
