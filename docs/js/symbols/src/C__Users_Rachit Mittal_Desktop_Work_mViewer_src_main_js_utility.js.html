  9     if (typeof String.prototype.format !== 'function') {

 12                 i;

 14                 formatted = formatted.replace("[" + i + "]", arguments[i]);
 15             }
 16             return formatted;
 17         };

 21             var str = this;
 22             if (!str || typeof str !== 'string') {
 23                 return null;
 24             }
 25             return str.replace(/^[\s]+/, '').replace(/[\s]+$/, '').replace(/[\s]{2,}/, ' ');
 26         };
 27     }

 42     MV.isArray = function (o) {
 43         return Object.prototype.toString.call(o) === '[object Array]';
 44     };
 45     MV.isObject = function (o) {
 46         return (typeof o === "object");
 47     };
 48     var getChildrenArray;
 49     getChildrenArray = function (doc) {
 50         var i, tempObject, key, value, newArray, childrenArray = [];
 51         var allKeys = MV.getProperties(doc);
 52         for (i = 0; i < allKeys.length; i++) {
 53             tempObject = newObject();
 54             key = allKeys[i];
 55             tempObject.key = key;
 56             value = doc[key];
 57             if (MV.isArray(value)) {
 58                 newArray = [];
 59                 //                    tempObject.type = "BasicDBList";
 60                 newArray = getChildrenArray(value);
 61                 tempObject.kiddies = newArray;
 62             } else if (MV.isObject(value)) {
 63                 newArray = [];
 64                 //                    tempObject.type = "BasicDBObject";
 65                 newArray = getChildrenArray(value);
 66                 tempObject.kiddies = newArray;
 67             } else {
 68                 tempObject.value = value;
 69             }
 70             childrenArray.push(tempObject);
 71         }
 72         return childrenArray;
 73     };
 74     var prepareReturnObject = function (response, result) {
 75             var returnObject = newObject();
 76             var resultObject = newObject();
 77             resultObject.results = result;
 78             returnObject.response = resultObject;
 79             returnObject.total_records = response.meta.totalRecords;
 80             returnObject.records_returned = response.meta.recordsReturned;
 81             returnObject.first_index = response.meta.startIndex;
 82             Y.log("Tree table data prepared", "info");
 83             return (returnObject);
 84         };
 85     MV.getTreebleDataforDocs = function (response) {
 86         var allDocs = response.results,
 87             aDoc, parentNode, childrenArray, result = [];
 88         var i;
 89         for (i = 0; i < allDocs.length; i++) {
 90             aDoc = allDocs[i];
 91             parentNode = newObject();
 92             childrenArray = [];
 93             parentNode.key = "Document [0]".format(i + 1);
 94             childrenArray = getChildrenArray(aDoc);
 95             parentNode.kiddies = childrenArray;
 96             result.push(parentNode);
 97         }
 98         return (prepareReturnObject(response, result));
 99     };
100     MV.getTreebleDataForServerStats = function (response) {
101         var data = response.results[0];
102         var aDoc, docCopy, parentNode, childrenArray, result = [],
103             finalObject, resultObject;
104         var i;
105         var allKeys = MV.getProperties(data);
106         for (i = 0; i < allKeys.length; i++) {
107             parentNode = newObject();
108             parentNode.key = allKeys[i];
109             var value = data[allKeys[i]];
110             if (MV.isObject(value)) {
111                 newArray = [];
112                 newArray = getChildrenArray(value);
113                 parentNode.kiddies = newArray;
114             } else {
115                 parentNode.value = value;
116             }
117             result.push(parentNode);
118         }
119         return (prepareReturnObject(response, result));
