 24     DS.prototype.parseJSONData = function (oRequest, oFullResponse) {
 25         var oParsedResponse = {
 26             results: [],
 27             meta: {}
 28         };
 29 
 30         if (lang.isObject(oFullResponse) && this.responseSchema.resultsList) {
 31             var schema = this.responseSchema,
 32                 fields = schema.fields,
 33                 resultsList = oFullResponse,
 34                 results = [],
 35                 metaFields = schema.metaFields || {},
 36                 fieldParsers = [],
 37                 fieldPaths = [],
 38                 simpleFields = [],
 39                 bError = false,
 40                 i, len, j, v, key, parser, path;
 41 
 42             // Function to convert the schema's fields into walk paths
 43             var buildPath = function (needle) {
 44                     var path = null,
 45                         keys = [],
 46                         i = 0;
 47                     if (needle) {
 48                         // Strip the ["string keys"] and [1] array indexes
 49                         needle = needle.
 50                         replace(/\[(['"])(.*?)\1\]/g, function (x, $1, $2) {
 51                             keys[i] = $2;
 52                             return '.@' + (i++);
 53                         }).
 54                         replace(/\[(\d+)\]/g, function (x, $1) {
 55                             keys[i] = parseInt($1, 10) | 0;
 56                             return '.@' + (i++);
 57                         }).
 58                         replace(/^\./, ''); // remove leading dot
 59                         // If the cleaned needle contains invalid characters, the
 60                         // path is invalid
 61                         if (!/[^\w\.\$@]/.test(needle)) {
 62                             path = needle.split('.');
 63                             for (i = path.length - 1; i >= 0; --i) {
 64                                 if (path[i].charAt(0) === '@') {
 65                                     path[i] = keys[parseInt(path[i].substr(1), 10)];
 66                                 }
 67                             }
 68                         }
 69 
 70                     }
 71                     return path;
 72                 };
 73 
 74 
 75             // Function to walk a path and return the pot of gold
 76             var walkPath = function (path, origin) {
 77                     var v = origin,
 78                         i = 0,
 79                         len = path.length;
 80                     for (; i < len && v; ++i) {
 81                         v = v[path[i]];
 82                     }
 83                     return v;
 84                 };
 85 
 86             // Parse the response
 87             // Step 1. Pull the resultsList from oFullResponse (default assumes
 88             // oFullResponse IS the resultsList)
 89             path = buildPath(schema.resultsList);
 90             if (path) {
 91                 resultsList = walkPath(path, oFullResponse);
 92                 if (resultsList === undefined) {
 93                     bError = true;
 94                 }
 95             } else {
 96                 bError = true;
 97             }
 98 
 99             if (!resultsList) {
100                 resultsList = [];
101             }
102 
103             if (!lang.isArray(resultsList)) {
104                 resultsList = [resultsList];
105             }
106 
107             if (!bError) {
108                 // Step 2. Parse out field data if identified
109                 if (schema.fields) {
110                     var field;
111                     // Build the field parser map and location paths
112                     for (i = 0, len = fields.length; i < len; i++) {
113                         field = fields[i];
114                         key = field.key || field;
115                         parser = ((typeof field.parser === 'function') ? field.parser : DS.Parser[field.parser]) || field.converter;
116                         path = buildPath(key);
117 
118                         if (parser) {
119                             fieldParsers[fieldParsers.length] = {
120                                 key: key,
121                                 parser: parser
122                             };
123                         }
124 
125                         if (path) {
126                             if (path.length > 1) {
127                                 fieldPaths[fieldPaths.length] = {
128                                     key: key,
129                                     path: path
130                                 };
131                             } else {
132                                 simpleFields[simpleFields.length] = {
133                                     key: key,
134                                     path: path[0]
135                                 };
136                             }
137                         }
138                     }
139 
140                     // Process the results, flattening the records and/or applying parsers if needed
141                     for (i = resultsList.length - 1; i >= 0; --i) {
142                         var r = resultsList[i],
143                             rec = {};
144                         if (r) {
145                             for (j = simpleFields.length - 1; j >= 0; --j) {
146                                 // Bug 1777850: data might be held in an array
147                                 rec[simpleFields[j].key] = (r[simpleFields[j].path] !== undefined) ? r[simpleFields[j].path] : r[j];
148                             }
149 
150                             for (j = fieldPaths.length - 1; j >= 0; --j) {
151                                 rec[fieldPaths[j].key] = walkPath(fieldPaths[j].path, r);
152                             }
153 
154                             for (j = fieldParsers.length - 1; j >= 0; --j) {
155                                 var p = fieldParsers[j].key;
156                                 rec[p] = fieldParsers[j].parser.call(this, rec[p]);
157                                 if (rec[p] === undefined) {
158                                     rec[p] = null;
159                                 }
160                             }
161                         }
162                         results[i] = rec;
163                     }
164                 } else {
165                     results = resultsList;
166                 }
167 
168                 for (key in metaFields) {
169                     if (lang.hasOwnProperty(metaFields, key)) {
170                         path = buildPath(metaFields[key]);
171                         if (path) {
172                             v = walkPath(path, oFullResponse);
173                             oParsedResponse.meta[key] = v;
174                         }
175                     }
176                 }
177 
178             } else {
179 
180                 oParsedResponse.error = true;
181             }
182 
183             oParsedResponse.results = results;
184         } else {
185             oParsedResponse.error = true;
186         }
187 
188         return oParsedResponse;
189     };

196     DT.prototype.load = function (oConfig) {
197         oConfig = oConfig || {};
198

203             scope: this,
204             argument: this.getState()
205         });

208     var origInitAttributes = DT.prototype.initAttributes;
209     DT.prototype.initAttributes = function () {
210         origInitAttributes.apply(this, arguments);
211 
212         /**
213          * @attribute displayAllRecords
214          * @description Set to true if you want to show all the records that were
215          * returned, not just the records that fall inside the paginator window.
216          * @type Boolean
217          * @default 0
218          */
219         this.setAttributeConfig("displayAllRecords", {
220             value: false,
221             validator: lang.isBoolean
222         });

225     /*
226      * Override to provide option to display all returned records, even if
227      * that is more than what paginator says is visible.
228      */
229     DT.prototype.render = function () {
230 
231         this._oChainRender.stop();
232 
233         this.fireEvent("beforeRenderEvent");
234 
235         var i, j, k, len, allRecords;
236 
237         var oPaginator = this.get('paginator');
238         // Paginator is enabled, show a subset of Records and update Paginator UI
239         if (oPaginator && this.get('displayAllRecords')) {
240             allRecords = this._oRecordSet.getRecords(
241             oPaginator.getStartIndex());
242         } else if (oPaginator) {
243             allRecords = this._oRecordSet.getRecords(
244             oPaginator.getStartIndex(), oPaginator.getRowsPerPage());
245         }
246         // Not paginated, show all records
247         else {
248             allRecords = this._oRecordSet.getRecords();
249         }
250 
251         // From the top, update in-place existing rows, so as to reuse DOM elements
252         var elTbody = this._elTbody,
253             loopN = this.get("renderLoopSize"),
254             nRecordsLength = allRecords.length;
255 
256         // Table has rows
257         if (nRecordsLength > 0) {
258             elTbody.style.display = "none";
259             while (elTbody.lastChild) {
260                 elTbody.removeChild(elTbody.lastChild);
261             }
262             elTbody.style.display = "";
263 
264             // Set up the loop Chain to render rows
265             this._oChainRender.add({
266                 method: function (oArg) {
267                     if ((this instanceof DT) && this._sId) {
268                         var i = oArg.nCurrentRecord,
269                             endRecordIndex = ((oArg.nCurrentRecord + oArg.nLoopLength) > nRecordsLength) ? nRecordsLength : (oArg.nCurrentRecord + oArg.nLoopLength),
270                             elRow, nextSibling;
271 
272                         elTbody.style.display = "none";
273 
274                         for (; i < endRecordIndex; i++) {
275                             elRow = Dom.get(allRecords[i].getId());
276                             elRow = elRow || this._addTrEl(allRecords[i]);
277                             nextSibling = elTbody.childNodes[i] || null;
278                             elTbody.insertBefore(elRow, nextSibling);
279                         }
280                         elTbody.style.display = "";
281 
282                         // Set up for the next loop
283                         oArg.nCurrentRecord = i;
284                     }
285                 },
286                 scope: this,
287                 iterations: (loopN > 0) ? Math.ceil(nRecordsLength / loopN) : 1,
288                 argument: {
289                     nCurrentRecord: 0,
290                     //nRecordsLength-1,  // Start at first Record
291                     nLoopLength: (loopN > 0) ? loopN : nRecordsLength
292                 },
293                 timeout: (loopN > 0) ? 0 : -1
294             });
295 
296             // Post-render tasks
297             this._oChainRender.add({
298                 method: function (oArg) {
299                     if ((this instanceof DT) && this._sId) {
300                         while (elTbody.rows.length > nRecordsLength) {
301                             elTbody.removeChild(elTbody.lastChild);
302                         }
303                         this._setFirstRow();
304                         this._setLastRow();
305                         this._setRowStripes();
306                         this._setSelections();
307                     }
308                 },
309                 scope: this,
310                 timeout: (loopN > 0) ? 0 : -1
311             });
312 
313         }
314         // Table has no rows
315         else {
316             // Set up the loop Chain to delete rows
317             var nTotal = elTbody.rows.length;
318             if (nTotal > 0) {
319                 this._oChainRender.add({
320                     method: function (oArg) {
321                         if ((this instanceof DT) && this._sId) {
322                             var i = oArg.nCurrent,
323                                 loopN = oArg.nLoopLength,
324                                 nIterEnd = (i - loopN < 0) ? -1 : i - loopN;
325 
326                             elTbody.style.display = "none";
327 
328                             for (; i > nIterEnd; i--) {
329                                 elTbody.deleteRow(-1);
330                             }
331                             elTbody.style.display = "";
332 
333                             // Set up for the next loop
334                             oArg.nCurrent = i;
335                         }
336                     },
337                     scope: this,
338                     iterations: (loopN > 0) ? Math.ceil(nTotal / loopN) : 1,
339                     argument: {
340                         nCurrent: nTotal,
341                         nLoopLength: (loopN > 0) ? loopN : nTotal
342                     },
343                     timeout: (loopN > 0) ? 0 : -1
344                 });
345             }
346         }
347         this._runRenderChain();
348     };
