 28 function LoadingPanel(message) {
 29 	// TODO: make loading panel module
 30     this.panel = new YAHOO.widget.Panel("wait", {
 31         width: "240px",
 32         fixedcenter: true,
 33         close: false,
 34         draggable: false,
 35         visible: false,
 36         constraintoviewport: true,
 37         iframe: true
 38     });
 39 
 40     this.panel.setHeader(message);
 41     this.panel.setBody("<img src='images/loading.gif'>");
 42     this.panel.render(document.getElementById("mainBody"));
 43 }
 44 LoadingPanel.prototype.show = function () {
 45     this.panel.show();
 46 };
 47 LoadingPanel.prototype.hide = function () {
 48     this.panel.hide();
 49 };
