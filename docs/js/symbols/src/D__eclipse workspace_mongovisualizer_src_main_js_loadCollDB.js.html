 33    var handleDropDB = function (target) {
 34          var request = Y.io("HomePopulate/dropDB?dbName=" + target,
 35          // configuration for dropping the database
 36          {
 37             method: "POST",
 38             on: {
 39                success: function (ioId, o) {
 40                   if (o.responseText === "success") {
 41                      alert("Success: Database Dropped");
 42                      window.location = "home.html";
 43                   } else {
 44                      alert("Unsuccessful: " + o.responseText);
 45                   }
 46                },
 47                failure: function (ioId, o) {
 48                   alert("Unsuccessful: " + o.statusText);
 49                }
 50             }
 51          });
 52          this.hide();
 53        };

 58    function onDBContextMenuClick(eventType, args) {
 59       var menuItem = args[1],// The MenuItem that was clicked
 60           target = this.contextEventTarget,
 61           dialog;
 62       switch (menuItem.index) {
 63       case 0:
 64          dialog = mongoVNamespace.getSimpleDialog("Do you really want to drop the Database?", handleDropDB(target.id), handleNo);
 65          targetNode = target.id;
 66 		 dialog.show();
 67 		 break;
 68       case 1:
 69          showDialog("collection", target.id);
 70          break;
 71       }
 72    }

 76    function handleGetDBSuccess(ioId, o) {
 77       Y.log("The success handler for loading DB called.", "info");
 78       if (o.responseText !== undefined) {
 79          var dbList = "",
 80              info, data, index, dbContextMenu;
 81          try {
 82             data = Y.JSON.parse(o.responseText);
 83          } catch (e) {
 84             alert("Cannot load Databases \nResponse Recieved :" + o.resposeText + "\nPlease check if MongoDB is running");
 85             window.location = "index.html";
 86          }
 87          for (index = 0; index < data.Name.length; index++) {
 88             dbList += "<li id='[0]' >[1]</li>".format(data.Name[index], data.Name[index]);
 89          }
 90          dbDiv.set("innerHTML", dbList);
 91          userName.set("innerHTML", data.UserName);
 92          hostName.set("innerHTML", data.Host);
 93          // Y.one("#addNewDB").set("innerHTML", "Create>>");
 94          loadingPanel.hide();
 95          dbContextMenu = new YAHOO.widget.ContextMenu("dbContextMenuID", {
 96             trigger: "dbNames",
 97             itemData: ["Delete Database", "Add Collection", "Statistics"]
 98          });
 99          dbContextMenu.render("dbContextMenu");
100          dbContextMenu.clickEvent.subscribe(onDBContextMenuClick);
101       }
102    }

196    function requestDBNames() {
197       loadingPanel = new LoadingPanel("Loading Databases...");
198       loadingPanel.show();
199       dbDiv = Y.one('#dbNames ul.lists');
200       userName = Y.one('#username');
201       hostName = Y.one('#hostname');
202       // dbDiv.set("innerHTML", "Loading database...");

215    function requestCollNames(e) {
216       mongoVNamespace.createDatatable("dbStats", Y.one("#mainBody"), Y.one("#mainBodyHeader"), e.currentTarget.get("id"));
217       Y.one('#queryForm').removeClass('form-cont');
218       loadingPanel = new LoadingPanel("Loading Collections...");
219       loadingPanel.show();
220       collDiv = Y.one('#collNames ul.lists');
221       // Y.one('#mainBodyHeader #options').set("innerHTML",
222       // "<span id='dropDB'>drop</span>");
223       Y.one('#queryForm').set("innerHTML", "");
224       mongoVNamespace.toggleClass(e.currentTarget, Y.all("#dbNames li"));
225       collDiv.set("innerHTML", "Loading collections...");
226       var request = Y.io("HomePopulate/coll?db=" + e.currentTarget.get("id"), {
227          // configuration for loading the collections
228          method: "POST",
229          on: {
230             success: handleGetCollSuccess,
231             failure: handleGetCollFailure
232          }
233       });
234       Y.log("Initiating request; Id: " + request.id + ".", "info");
235    }

241    function showDialog(item, parent) {
242       var form, dialog;
243       var handleAddDBSuccess = function (o) {
244             var data = Y.JSON.parse(o.responseText);
245             if (data.result === "success") {
246                alert("Creation Successful");
247                window.location = "home.html";
248             } else {
249                alert("Creation failed:\n " + data.result);
250             }
251           };
252       var handleAddCollSuccess = function (o) {
253             var data = Y.JSON.parse(o.responseText);
254             if (data.result === "success") {
255                alert("Creation Successful");
256                Y.one("#" + data.db).simulate("click");
257             } else {
258                alert("Creation failed:\n " + data.result);
259             }
260           };
261       var handleFailure = function (o) {
262             alert("Creation failed: " + o.status);
263           };
264       if (item === "database") {
265          form = "addDBDialog";
266          dialog = new Dialog(form, handleAddDBSuccess, handleFailure);
267       } else if (item === "collection") {
268          form = "addColDialog";
269          dialog = new Dialog(form, handleAddCollSuccess, handleFailure);
270       }
271       dialog.show();
272    }

285          } else if (menuOpt === "collStats") {
286             mongoVNamespace.createDatatable("collStats", Y.one("#mainBody"), Y.one("#mainBodyHeader"), e.currentTarget.get("id"));
287             header.set("innerHTML", "Collection Statistics");
288          } else if (menuOpt === 'chart') {
289             header.set("innerHTML", "");
290             window.open('chart2.html', '_newtab');
291          }
292        }; /* EVENT LISTENERS */
