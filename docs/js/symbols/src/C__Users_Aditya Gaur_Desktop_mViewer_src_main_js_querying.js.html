 16 YUI({
 17     filter: 'raw'
 18 }).use("yes-no-dialog", "alert-dialog", "io-base", "json-parse", "node-event-simulate", "node", "event-delegate", "stylize", "json-stringify", "utility", function (Y) {
 19     Y.namespace('com.imaginea.mongoV');
 20     var MV = Y.com.imaginea.mongoV;
 21     MV.treebleData = {};
 22     var showTabView = function (e) {
 23             var treeble;
 24             MV.toggleClass(e.currentTarget, Y.all("#collNames li"));
 25             Y.one("#currentColl").set("value", e.currentTarget.get("id"));
 26             MV.mainBody.empty(true);
 27             var getQueryParameters = function () {
 28                     var parsedQuery, query = Y.one('#queryBox').get("value");
 29                     var limit = Y.one('#limit').get("value");
 30                     var skip = Y.one('#skip').get("value");
 31                     var fields = Y.all('#fields input');
 32                     var index = 0;
 33                     if (query === "") {
 34                         query = "{}";
 35                     }
 36                     query = query.replace(/'/g, '"');
 37                     var checkedFields = [];
 38                     try {
 39                         parsedQuery = Y.JSON.parse(query);
 40                         for (index = 0; index < fields.size(); index++) {
 41                             var item = fields.item(index);
 42                             if (item.get("checked")) {
 43                                 checkedFields.push(item.get("name"));
 44                             }
 45                         }
 46                         return ("&limit=[0]&skip=[1]&fields=[2]&query=[3]".format(limit, skip, checkedFields, query));
 47                     } catch (error) {
 48                         Y.log("Could not parse query. Reason: [0]".format(error), "error");
 49                         MV.showAlertDialog("Failed:Could not parse query. [0]".format(error), MV.warnIcon);
 50                     }
 51                 };
 52             var defineDatasource = function () {
 53                     MV.data = new YAHOO.util.XHRDataSource(MV.URLMap.getDocs(), {
 54                         responseType: YAHOO.util.XHRDataSource.TYPE_JSON,
 55                         responseSchema: {
 56                             resultsList: "response.result",
 57                             metaFields: {
 58                                 startIndex: 'first_index',
 59                                 recordsReturned: 'records_returned',
 60                                 totalRecords: 'total_records'
 61                             }
 62                         }
 63                     });
 64                 };
 65             var sendDocumentRequest = function (param) {
 66                     MV.data.sendRequest(param, {
 67                         success: showDocuments,
 68                         failure: function (request, responseObject) {
 69                             MV.showAlertDialog("Failed: Documents could not be loaded", MV.warnIcon);
 70                             Y.log("Documents could not be loaded. Response: [0]".format(responseObject.responseText), "error");
 71                         },
 72                         scope: tabView
 73                     });
 74                 };
 75             var executeQuery = function (e) {
 76                     var queryParams = getQueryParameters();
 77                     if (queryParams !== undefined) {
 78                         sendDocumentRequest(queryParams);
 79                     }
 80                 };
 81             var showQueryBox = function (ioId, responseObject) {
 82                     Y.log("Preparing to show QueryBox", "info");
 83                     try {
 84                         Y.log("Parsing the JSON response to get the keys", "info");
 85                         var parsedResponse = Y.JSON.parse(responseObject.responseText);
 86                         var keys = parsedResponse.response.result;
 87                         if (keys !== undefined) {
 88                             var queryForm = Y.one('#queryForm');
 89                             queryForm.addClass('form-cont');
 90                             queryForm.set("innerHTML", MV.getForm(keys));
 91                             Y.log("QueryBox loaded", "info");
 92                             Y.on("click", executeQuery, "#execQueryButton");
 93                             defineDatasource();
 94                             sendDocumentRequest(getQueryParameters());
 95                         } else {
 96                             var error = parsedResponse.response.error;
 97                             Y.log("Could not get keys. Message: [0]".format(error.message), "error");
 98                             MV.showAlertDialog("Could not load the query Box! [0]".format(MV.errorCodeMap(error.code)), MV.warnIcon);
 99                         }
100                     } catch (e) {
101                         Y.log("Could not parse the JSON response to get the keys", "error");
102                         Y.log("Response received: [0]".format(responseObject.resposeText), "error");
103                         MV.showAlertDialog("Cannot parse Response to get keys!", MV.warnIcon);
104                     }
105                 };
106             var getKeyRequest = Y.io(MV.URLMap.documentKeys(), {
107                 method: "GET",
108                 on: {
109                     success: showQueryBox,
110                     failure: function (ioId, responseObject) {
111                         MV.showAlertDialog("Unexpected Error: Could not load the query Box", MV.warnIcon);
112                         Y.log("Could not send the request to get the keys in the collection. Response Status: [0]".format(responseObject.statusText), "error");
113                     }
114                 }
115             });
116             var fitToContent = function (maxHeight, text) {
117                     var adjustedHeight = text.clientHeight;
118                     if (!maxHeight || maxHeight > adjustedHeight) {
119                         adjustedHeight = Math.max(text.scrollHeight, adjustedHeight);
120                         if (maxHeight) {
121                             adjustedHeight = Math.min(maxHeight, adjustedHeight);
122                         }
123                         if (adjustedHeight > text.clientHeight) {
124                             text.style.height = adjustedHeight + "px";
125                         }
126                     }
127                 };
128             var loadAndSubscribe = function (treeble) {
129                     treeble.load();
130                     treeble.subscribe("rowMouseoverEvent", treeble.onEventHighlightRow);
131                     treeble.subscribe("rowMouseoutEvent", treeble.onEventUnhighlightRow);
132                 };
133             var showDocuments = function (request, responseObject) {
134                     Y.log("Preparing to write on JSON tab", "info");
135                     writeOnJSONTab(responseObject.results);
136                     Y.log("Preparing the treeTable data", "info");
137                     var treebleData = MV.getTreebleDataforDocs(responseObject);
138                     treeble = MV.getTreeble(treebleData);
139                     loadAndSubscribe(treeble);
140                     Y.log("Tree table view loaded", "info");
141                 };
142             var tabView = new YAHOO.widget.TabView();
143             tabView.addTab(new YAHOO.widget.Tab({
144                 label: 'JSON',
145                 cacheData: true,
146                 active: true
147             }));
148             tabView.addTab(new YAHOO.widget.Tab({
149                 label: 'Tree Table',
150                 content: ' <div id="table"></div><div id="table-pagination"></div> '
151             }));
152             var actionMap = {
153                 save: "save",
154                 edit: "edit"
155             };
156             var idMap = {};
157             var getButtonIndex = function (targetNode) {
158                     var btnID = targetNode.get("id");
159                     var match = btnID.match(/\d+/);
160                     return (parseInt(match[0], 10));
161                 };
162             var toggleSaveEdit = function (targetNode, index, action) {
163                     var textArea = Y.one('#doc' + index).one("pre").one("textarea");
164                     var antiAction;
165                     if (action === actionMap.save) {
166                         antiAction = actionMap.edit;
167                         textArea.addClass('disabled');
168                         textArea.setAttribute("disabled", "disabled");
169                         Y.on("click", editDoc, "#edit" + index);
170                     } else {
171                         antiAction = actionMap.save;
172                         textArea.removeAttribute("disabled");
173                         textArea.removeClass('disabled');
174                         Y.on("click", saveDoc, "#save" + index);
175                     }
176                     targetNode.set("innerHTML", antiAction);
177                     targetNode.removeClass(action + 'btn');
178                     targetNode.addClass(antiAction + 'btn');
179                     targetNode.set("id", antiAction + index);
180                 };
181             var parseUpdateDocResponse = function (ioId, responseObject) {
182                     var parsedResponse = Y.JSON.parse(responseObject.responseText);
183                     response = parsedResponse.response.result;
184                     if (response !== undefined) {
185                         MV.showAlertDialog("Document updated", MV.infoIcon);
186                         Y.log("Document update to [0]".format(response), "info");
187                         Y.one("#" + Y.one("#currentColl").get("value")).simulate("click");
188                     } else {
189                         var error = parsedResponse.response.error;
190                         MV.showAlertDialog("Could not update Document! [0]".format(MV.errorCodeMap[error.code]), MV.warnIcon, function () {
191                             Y.one("#" + Y.one("#currentColl").get("value")).simulate("click");
192                         });
193                         Y.log("Could not update Document! [0]".format(MV.errorCodeMap[error.code]), "error");
194                     }
195                 };
196             var sendUpdateDocRequest = function (doc, id) {
197                     var updateDocumentRequest = Y.io(MV.URLMap.updateDoc(), {
198                         method: "POST",
199                         data: "_id=" + id + "&keys=" + doc,
200                         on: {
201                             success: parseUpdateDocResponse,
202                             failure: function (ioId, responseObject) {
203                                 MV.showAlertDialog("Unexpected Error: Could not update the document. Check if app server is running", MV.warnIcon);
204                                 Y.log("Could not send the request to update the document. Response Status: [0]".format(responseObject.statusText), "error");
205                             }
206                         }
207                     });
208                 };
209             var allKeysSelected = function () {
210                     var fields = Y.all('#fields input');
211                     var index;
212                     for (index = 0; index < fields.size(); index++) {
213                         var item = fields.item(index);
214                         if (!item.get("checked")) {
215                             return false;
216                         }
217                     }
218                     return true;
219                 };
220             var selectAllKeys = function () {
221                     var fields = Y.all('#fields input');
222                     var index;
223                     for (index = 0; index < fields.size(); index++) {
224                         var item = fields.item(index);
225                         item.set("checked", "true");
226                     }
227                     executeQuery();
228                     this.hide();
229                 };
230             var deleteDoc = function (eventObject) {
231                     var sendDeleteDocRequest = function () {
232                             var targetNode = eventObject.currentTarget;
233                             var index = getButtonIndex(targetNode);
234                             var doc = Y.one('#doc' + index).one("pre").one("textarea").get("value");
235                             parsedDoc = Y.JSON.parse(doc);
236                             var id = parsedDoc._id;
237                             var request = Y.io(MV.URLMap.deleteDoc(),
238                             // configuration for dropping the document
239                             {
240                                 method: "POST",
241                                 data: "_id=" + id,
242                                 on: {
243                                     success: function (ioId, responseObj) {
244                                         var parsedResponse = Y.JSON.parse(responseObj.responseText);
245                                         response = parsedResponse.response.result;
246                                         if (response !== undefined) {
247                                             MV.showAlertDialog("Document deleted", MV.infoIcon);
248                                             Y.log("Document with _id= [0] deleted. Response: [1]".format(id, response), "info");
249                                             Y.one("#" + Y.one("#currentColl").get("value")).simulate("click");
250                                         } else {
251                                             var error = parsedResponse.response.error;
252                                             MV.showAlertDialog("Could not delete the document with _id [0]. [1]".format(id, MV.errorCodeMap[error.code]), MV.warnIcon);
253                                             Y.log("Could not delete the document with _id =  [0], Error message: [1], Error Code: [2]".format(id, error.message, error.code), "error");
254                                         }
255                                     },
256                                     failure: function (ioId, responseObj) {
257                                         Y.log("Could not delete the document .Status text: ".format(Y.one("#currentColl").get("value"), responseObj.statusText), "error");
258                                         MV.showAlertDialog("Could not drop the document! Please check if your app server is running and try again. Status Text: [1]".format(responseObj.statusText), MV.warnIcon);
259                                     }
260                                 }
261                             });
262                             this.hide();
263                         };
264                     MV.showYesNoDialog("Do you really want to drop the document ?", sendDeleteDocRequest, function () {
265                         this.hide();
266                     });
267                 };
268             var saveDoc = function (eventObject) {
269                     var parsedDoc;
270                     var targetNode = eventObject.currentTarget;
271                     var index = getButtonIndex(targetNode);
272                     toggleSaveEdit(targetNode, index, actionMap.save);
273                     var doc = Y.one('#doc' + index).one("pre").one("textarea").get("value");
274                     doc = doc.replace(/'/g, '"');
275                     try {
276                         parsedDoc = Y.JSON.parse(doc);
277                     } catch (e) {
278                         MV.showAlertDialog("The document entered is not in the correct JSON format");
279                     }
280                     sendUpdateDocRequest(Y.JSON.stringify(parsedDoc), idMap.index);
281                 };
282             var editDoc = function (eventObject) {
283                     if (!allKeysSelected()) {
284                         MV.showYesNoDialog("To edit a document you need check all keys in query box. Click YES to do so, NO to cancel", selectAllKeys, function () {
285                             this.hide();
286                         });
287                     } else {
288                         var targetNode = eventObject.currentTarget;
289                         var index = getButtonIndex(targetNode);
290                         toggleSaveEdit(targetNode, index, actionMap.edit);
291                         var doc = Y.one('#doc' + index).one("pre").one("textarea").get("value");
292                         parsedDoc = Y.JSON.parse(doc);
293                         idMap.index = parsedDoc._id;
294                     }
295                 };
296             var writeOnJSONTab = function (response) {
297                     var documents = "<table>",
298                         i;
299                     for (i = 0; i < response.length; i++) {
300                         documents = documents + "<tr><td id='doc" + i + "'><pre><textarea id='ta" + i + "' class='disabled' disabled='disabled' cols='80' >" + Y.JSON.stringify(response[i], null, 4) + "</textarea></pre></td>" + "<td><button id='edit" + i + "' class='btn editbtn'>Edit</button></td>" + "<td><button id='delete" + i + "' class='btn deletebtn'>delete</button></td></tr>";
301                     }
302                     if (i === 0) {
303                         documents = documents + "No documents to be displayed";
304                     }
305                     documents = documents + "</table>";
306                     tabView.getTab(0).setAttributes({
307                         content: documents
308                     }, false);
309                     for (i = 0; i < response.length; i++) {
310                         Y.on("click", editDoc, "#edit" + i);
311                         Y.on("click", deleteDoc, "#delete" + i);
312                     }
313                     for (i = 0; i < response.length; i++) {
314                         fitToContent(500, document.getElementById("ta" + i));
315                     }
316                     Y.log("The documents written on the JSON tab", "info");
317                 };
