 25 // TODO send a custom handler if not provided run the default one
 26 YUI.add('alert-dialog', function (Y) {
 27     Y.namespace('com.imaginea.mongoV');
 28     Y.com.imaginea.mongoV.showAlertDialog = function (msg, ico, handler) {
 29         var simpleDialog = new YAHOO.widget.SimpleDialog("simpledialog", {
 30             width: "300px",
 31             fixedcenter: true,
 32             visible: false,
 33             draggable: true,
 34             close: false,
 35             text: msg,
 36             icon: ico || YAHOO.widget.SimpleDialog.ICON_INFO,
 37             constraintoviewport: true,
 38             zIndex : 10,
 39             buttons: [{
 40                 text: "OK",
 41                 handler: handler ||
 42                 function () {
 43                     this.hide();
 44                 },
 45                 isDefault: true
 46             }]
 47         });
 48         simpleDialog.setHeader("Info");
 49         simpleDialog.render("alertDialogContainer");
 50         simpleDialog.show();
 51     };
 52 }, '3.3.0', {
 53     requires: []
 54 });
