 25 YUI({
 26     filter: 'raw'
 27 }).use("alert-dialog", "utility", "node", "event-delegate", "stylize", "custom-datatable", function (Y) {
 28     Y.namespace('com.imaginea.mongoV');
 29     var MV = Y.com.imaginea.mongoV;
 30     var showServerStats = function () {
 31             var data = new YAHOO.util.XHRDataSource(MV.URLMap.serverStatistics(), {
 32                 responseType: YAHOO.util.XHRDataSource.TYPE_JSON,
 33                 responseSchema: {
 34                     resultsList: "response.result",
 35                     metaFields: {
 36                         startIndex: 'first_index',
 37                         recordsReturned: 'records_returned',
 38                         totalRecords: 'total_records'
 39                     }
 40                 }
 41             });
 42             data.sendRequest("", {
 43                 success: function (request, responseObject) {
 44                     MV.mainBody.set("innerHTML", '<div id="table"></div><div id="table-pagination"></div>');
 45                     var treebleData = MV.getTreebleDataForServerStats(responseObject);
 46                     treeble = MV.getTreeble(treebleData);
 47                     treeble.load();
 48                 },
 49                 failure: function (request, responseObject) {
 50                     MV.showAlertDialog("Failed: ServerStats could not be loaded! Please check if the app server is still running.", MV.warnIcon);
 51                     Y.log("Server Statistics could not be loaded. Response Status: [0]".format(responseObject.statusText), "error");
 52                 },
 53             });
 54         };
 55     var executeHeaderOption = function (eventObject) {
 56             menuOpt = eventObject.currentTarget.get("id");
 57             MV.toggleClass(eventObject.currentTarget, Y.all(".nav-cont li a"));
 58             // Clearing the content of main Body
 59             MV.mainBody.set("innerHTML", "");
 60             Y.one('#queryForm').set("innerHTML", "");
 61             if (menuOpt === "home") {
 62                 window.location = "home.html?tokenID=" + Y.one("#tokenID").get("value") + "&username=" + Y.one("#username").get("value") + "&host=" + Y.one("#host").get("value");
 63             } else if (menuOpt === "serverStats") {
 64                 showServerStats();
 65                 MV.header.set("innerHTML","Server Statistics");
 66             } else if (menuOpt === 'graphs') {
 67                 MV.header.set("innerHTML", "");
 68                 window.open(MV.URLMap.graphs(), '_newtab');
 69             }
 70         };
 71     var logout = function (eventObject) {
 72             var request = Y.io(MV.URLMap.logout(), {
 73                 method: "GET",
 74                 on: {
 75                     success: function (ioId, responseObject) {
 76                         parsedResponse = Y.JSON.parse(responseObject.responseText);
 77                         var response = parsedResponse.response.result;
 78                         if (response !== undefined) {
 79                             Y.log("Successfully logging out.", "info");
 80                             window.location = "loggedOut.html";
 81                         } else {
 82                             var error = parsedResponse.response.error;
 83                             MV.showAlertDialog("Cannot logout! [0]".format(MV.errorCodeMap[error.code]), MV.warnIcon);
 84                             Y.log("Could not logout. Message: [0], Code: [1]".format(error.message, error.code), "error");
 85                         }
 86                     },
 87                     failure: function (ioId, responseObject) {
 88                         MV.showAlertDialog("Could not send request. Check if app server is running. Response message: [0]".format(responseObject.statusText), MV.warnIcon);
 89                         Y.log("Could not send request to logout. Status text: [0] ".format(responseObject.statusText), "error");
 90                     }
 91                 }
 92             });
 93         };
 94     Y.delegate("click", executeHeaderOption, ".nav-cont", "li a");
 95     Y.on("click", logout, "#logout");
 96 });
