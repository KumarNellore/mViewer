 25 YUI().use("io-base", "node-base", "json-parse", function (Y) {
 26 
 27     function executeQuery(e) {
 28         var data, query = Y.one('#queryBox').get("value"),
 29             limit = Y.one('#limit').get("value"),
 30             skip = Y.one('#skip').get("value"),
 31             fields = Y.all('#fields input'),
 32             index = 0;
 33         var checkedFields = [];
 34         try {
 35 
 36             data = Y.JSON.parse(query);
 37             for (index = 0; index < fields.size(); index++) {
 38                 var item = fields.item(index);
 39                 if (item.get("checked")) {
 40                     checkedFields.push(item.get("name"));
 41                 }
 42             }
 43 
 44             node = Y.one("#mainBody");
 45             header = Y.one("#mainBodyHeader #title");
 46             coll = e.currentTarget.get("id");
 47             node.set("innerHTML", "");
 48             var tabView = new YAHOO.widget.TabView();
 49             tabView.addTab(new YAHOO.widget.Tab({
 50                 label: 'JSON',
 51                 dataSrc: 'HomePopulate/queriedDocs?type=json' + '&limit=' + limit + '&skip=' + skip + '&fields=' + checkedFields + '&query=' + query,
 52                 cacheData: true,
 53                 active: true
 54             }));
 55             tabView.addTab(new YAHOO.widget.Tab({
 56                 label: 'Treeble',
 57                 content: ' <div id="table"></div><div id="table-pagination"></div> '
 58             }));
 59             tabView.addTab(new YAHOO.widget.Tab({
 60                 label: 'TEXT',
 61                 dataSrc: 'HomePopulate/queriedDocs?type=text' + '&limit=' + limit + '&skip=' + skip + '&fields=' + checkedFields + '&query=' + query,
 62                 cacheData: true
 63             }));
 64 
 65             YAHOO.log("The example has finished loading; as you interact with it, you'll see log messages appearing here.", "info", "example");
 66 
 67             tabView.appendTo('mainBody');
 68             tabView.getTab(1).addListener("click", function () {
 69                 YAHOO.namespace('treetable');
 70                 var Dom = YAHOO.util.Dom,
 71                     Event = YAHOO.util.Event,
 72                     DT = YAHOO.widget.DataTable;
 73 
 74                 function saveAndValidate(
 75                 inputValue, currentValue, editorInstance) {
 76                     alert();
 77                 }
 78 
 79                 function xhrGenerateRequest(
 80                 state, path) {
 81 
 82                     return "startIndex=" + state.startIndex + (state.results !== null ? "&results=" + state.results : "") + (state.sort ? "&sort=" + state.sort : "") + (state.dir ? "&dir=" + state.dir : "") + (path && path.length > 0 ? "&kiddies=" + path.join("-") : "");
 83                 }
 84 
 85                 var myDataSource = new YAHOO.util.XHRDataSource('HomePopulate/getTreebleData?' + 'limit=' + limit + '&skip=' + skip + '&fields=' + checkedFields + '&query=' + query + '&', {
 86                     responseType: YAHOO.util.XHRDataSource.TYPE_JSON,
 87                     responseSchema: {
 88                         resultsList: "Response.Results",
 89                         fields: ["key", "value", "type",
 90                         {
 91                             key: 'kiddies',
 92                             parser: 'datasource'
 93                         }],
 94                         metaFields: {
 95                             startIndex: 'first_index',
 96                             recordsReturned: 'records_returned',
 97                             totalRecords: 'total_records'
 98                         }
 99 
100                     },
101                     treebleConfig: {
102                         childNodesKey: 'kiddies',
103                         generateRequest: xhrGenerateRequest,
104                         totalRecordsReturnExpr: '.meta.recordsReturned',
105                         startIndexExpr: '.meta.startIndex'
106                     }
107                 });
108                 
109 
110                 YAHOO.treetable.localDataTable = new YAHOO.widget.DataTable(
111                 // Root element id
112                 "table",
113                 // Column configuration
114                 [{
115                     key: "toggle_column",
116                     label: "",
117                     formatter: function (
118                     elCell, oRecord, oColumn, oData) {
119                         Dom.addClass(
120                         elCell.parentNode, 'treeble-nub');
121                         if (oRecord.getData('kiddies')) {
122                             var path = oRecord.getData('_yui_node_path');
123                             var open = this.rowIsOpen(path);
124                             var clazz = open ? 'row-open' : 'row-closed';
125 
126                             Dom.addClass(
127                             elCell, 'row-toggle');
128                             Dom.replaceClass(
129                             elCell, /row-(open|closed)/, clazz);
130                             elCell.innerHTML = '<a class="treeble-collapse-nub" href="javascript:void(0);"></a>';
131 
132                             Event.on(
133                             elCell, 'click', function (
134                             e, path) {
135                                 this.toggleRow(path);
136                             }, path, this);
137                         }
138                     }
139                 }, {
140                     key: "key",
141                     label: "Key",
142                     width: node.get('scrollWidth') / 3 - 38,
143                     resizeable: true,
144                     sortable: true,
145                     formatter: function (
146                     elCell, oRecord, oColumn, oData) {
147                         elCell.innerHTML = '<span style="font-weight: bolder;padding-left:' + oRecord.getData('_yui_node_depth') * 15 + 'px;">' + oData + '</span>';
148                     }
149                 }, {
150                     key: "value",
151                     label: "Value",
152                     width: node.get('scrollWidth') / 3 - 38,
153                     resizeable: true,
154                     editor: new YAHOO.widget.TextboxCellEditor()
155                 }, {
156                     key: "type",
157                     label: "Type",
158                     width: node.get('scrollWidth') / 3 - 38,
159                     resizeable: true
160                 }],
161 
162                 // Data Source
163                 new YAHOO.util.TreebleDataSource(
164                 myDataSource, {
165                     paginateChildren: false
166                 }),
167                 // Attribute Config
168                 {
169                     paginator: new YAHOO.widget.Paginator({
170                         rowsPerPage: limit,
171                         // totalRecords:oResponse.meta.totalRecords,
172                         rowsPerPageOptions: [
173                         1, 2, 5, 10, 25],
174                         containers: 'table-pagination',
175                         template: '{FirstPageLink}{PreviousPageLink}{PageLinks}{NextPageLink}{LastPageLink}{RowsPerPageDropdown}',
176                         pageLinks: 5
177                     }),
178                     initialLoad: false,
179                     initialRequest: {
180                         startIndex: skip,
181                         results: limit
182                     },
183                     dynamicData: true,
184                     displayAllRecords: true,
185                     generateRequest: DT.generateTreebleDataSourceRequest
186 
187                 });
188                 YAHOO.treetable.localDataTable.handleDataReturnPayload = function (
189                 oRequest, oResponse, oPayload) {
190                     oPayload.totalRecords = oResponse.meta.totalRecords;
191                     return oPayload;
192                 };
193 
194                 YAHOO.treetable.localDataTable.load();
195 
196                 // Cell editing
197                 var editors = {
198                     String: new YAHOO.widget.TextboxCellEditor(),
199                     Double: new YAHOO.widget.TextboxCellEditor({
200                         validator: function (
201                         val) {
202                             val = parseFloat(val);
203                             if (YAHOO.lang.isNumber(val)) {
204                                 return val;
205                             }
206                         }
207                     }),
208                     Integer: new YAHOO.widget.TextboxCellEditor({
209                         validator: function (
210                         val) {
211                             val = parseFloat(val);
212                             if (YAHOO.lang.isNumber(val)) {
213                                 return val;
214                             }
215                         }
216                     }),
217                     Date: new YAHOO.widget.DateCellEditor()
218 
219                 };
220 
221                 YAHOO.treetable.localDataTable.subscribe("cellClickEvent", function (oArgs) {
222                     var target = oArgs.target;
223                     var record = this.getRecord(target);
224                     var column = this.getColumn(target);
225                     if (column.key === 'value') {
226 
227                         var type = record.getData('type');
228                         column.editor = editors[type];
229                         this.showCellEditor(target);
230 
231                     }
232                 });
233                 YAHOO.treetable.localDataTable.subscribe("cellMouseoverEvent", function (oArgs) {
234                     var target = oArgs.target;
235 
236                     var column = this.getColumn(target);
237                     if (column.key === 'value') {
238                         this.highlightCell(target);
239                     }
240                 });
241 
242                 function saveValue(oArgs) {
243                     // This is a rather strange
244                     // thing to do, but it
245                     // demonstrates
246                     // that we can get the
247                     // DataSource and modify the
248                     // value. An XHR
249                     // request would make more
250                     // sense, but the data
251                     // happens to be local.
252                     var rec = oArgs.editor.getRecord();
253                     var ds = rec.getData('_yui_node_ds');
254                     var path = rec.getData('_yui_node_path');
255                     ds.liveData[path[path.length - 1]].quantity = oArgs.newData;
256                 }
257                 YAHOO.treetable.localDataTable.subscribe("cellMouseoutEvent", YAHOO.treetable.localDataTable.onEventUnhighlightCell);
258                 YAHOO.treetable.localDataTable.subscribe("editorSaveEvent", saveValue);
259             });
260 
261         } catch (error) {
262             alert("Error: " + error);
263         }
264 
265     }
266 
267     Y.on("click", executeQuery, "#execQueryButton");
268 });
