 16 YUI.add('dialog-box', function (Y) {
 17     Y.namespace('com.imaginea.mongoV');
 18     var MV = Y.com.imaginea.mongoV;
 19     MV.getDialog = function Dialog(form, successHandler, failureHandler) {
 20         YAHOO.util.Dom.removeClass(form, "yui-pe-content");
 21         var handleCancel = function () {
 22                 this.cancel();
 23             };
 24         var addColDialoghandleSubmit = function () {
 25                 Y.log("Submit handler for add collection called","info");
 26                 var newCollInfo = this.getData();
 27                 if (newCollInfo.name === "") {
 28                     MV.showAlertDialog("Please enter the name.");
 29                 } else {
 30                     Y.one("#newName").set("value", newCollInfo.name);
 31                     Y.one("#" + form + " .bd form").setAttribute("action", MV.URLMap.insertColl());
 32                     this.submit();
 33                 }
 34             };
 35         var addDBDialoghandleSubmit = function () {
 36                 var newDBInfo = this.getData();
 37                 if (newDBInfo.name === "") {
 38                     MV.showAlertDialog("Please enter the name.");
 39                 } else {
 40                     Y.one("#newName").set("value", newDBInfo.name);
 41                     Y.one("#" + form + " .bd form").setAttribute("action", MV.URLMap.insertDB());
 42                     this.submit();
 43                 }
 44             };
 45         var addDocDialoghandleSubmit = function () {
 46                 var newDoc = this.getData().document;
 47                 try {
 48                     Y.JSON.parse(newDoc);
 49                     Y.one("#" + form + " .bd form").setAttribute("action", MV.URLMap.insertDoc());
 50                     this.submit();
 51                 } catch (e) {
 52                     MV.showAlertDialog("Please enter the new document in JSON format", MV.warnIcon);
 53                     Y.log("New Document format not JSON", "error");
 54                 }
 55             };
 56         var sumbitHandlerMap = {
 57             "addColDialogSubmitHandler": addColDialoghandleSubmit,
 58             "addDBDialogSubmitHandler": addDBDialoghandleSubmit,
 59             "addDocDialogSubmitHandler": addDocDialoghandleSubmit
 60         };
 61         var dialogBox = new YAHOO.widget.Dialog(form, {
 62             width: "30em",
 63             fixedcenter: true,
 64             visible: false,
 65             effect: {
 66                 effect: YAHOO.widget.ContainerEffect.SLIDE,
 67                 duration: 0.25
 68             },
 69             constraintoviewport: true,
 70             buttons: [{
 71                 text: "Submit",
 72                 handler: sumbitHandlerMap[form + "SubmitHandler"] ||
 73                 function () {
 74                     this.submit();
 75                 },
 76                 isDefault: true
 77             }, {
 78                 text: "Cancel",
 79                 handler: handleCancel
 80             }]
 81         });
 82         dialogBox.callback = {
 83             success: successHandler,
 84             failure: failureHandler
 85         };
 86         dialogBox.render();
 87         dialogBox.show();
 88         return dialogBox;
 89     };
 90 }, '3.3.0', {
 91     requires: ["utility", "node", "alert-dialog"]
 92 });
