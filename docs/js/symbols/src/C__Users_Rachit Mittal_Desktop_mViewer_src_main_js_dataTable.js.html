 17 YUI.add('custom-datatable', function(Y) {
 18     Y.namespace('com.imaginea.mongoV');
 19     /**
 20      * <p>custom-datatable creates a datatable with the following columns
 21      * <p>Currently it is used to make the datatable to show the collection and the
 22      * database statistics. </p>
 23      *
 24      * @module custom-datatable
 25      * @namespace com.imaginea.mongov
 26      * @requires "io-base", "node", "json-parse", "datatable-scroll", "datasource-io", "datasource-jsonschema", "datatable-datasource", "event-delegate"
 27      * @param path
 28      *          <dd>(required) This is the url to which the request will be sent to get the data</dd>
 29      * @param name
 30      *          <dd>(required)The name of the collection/database whose data is required.</dd>
 31      */
 32     var MV = Y.com.imaginea.mongoV;
 33     MV.createDatatable = function(path, name) {
 34         MV.mainBody.set("innerHTML", "");
 35         /**
 36          * <p>The function is used to make the column configuration for the datatable  
 37          * @param colKey
 38          * <dd>(required)The column name</dd>
 39          **/
 40         var createColumn = function(colKey) {
 41             return {
 42                 key: colKey,
 43                 sortable: true,
 44                 width: "210px"
 45             };
 46         };
 47         var cols = [createColumn("Key"), createColumn("Value"), createColumn("Type")];
 48         var ds = new Y.DataSource.IO({
 49             source: path
 50         });
 51         ds.plug(Y.Plugin.DataSourceJSONSchema, {
 52             schema: {
 53                 resultListLocator: "response.result",
 54                 resultFields: ["Key", "Value", "Type"]
 55             }
 56         });
 57         var dt = new Y.DataTable.Base({
 58             columnset: cols,
 59             width: "697px"
 60         }).plug(Y.Plugin.DataTableDataSource, {
 61             datasource: ds,
 62             initialRequest: ""
 63         });
 64 
 65         ds.after("response", function() {
 66 
 67             MV.header.addClass('tab-cont');
 68             MV.header.set("innerHTML", "Statistics: " + name);
 69             dt.render("#" + MV.mainBody.get('id'));
 70         });
 71     };
 72 }, '3.3.0', {
 73     requires: ["io-base", "node", "json-parse", "datatable-scroll", "datasource-io", "datasource-jsonschema", "datatable-datasource", "event-delegate"]
 74 });
