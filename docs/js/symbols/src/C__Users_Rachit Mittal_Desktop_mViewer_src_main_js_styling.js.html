 16 YUI.add('stylize', function (Y) {
 17     Y.namespace('com.imaginea.mongoV');
 18     var MV = Y.com.imaginea.mongoV;
 19     /**
 20      * <p>toggleClass adds a class to the menu option as soon as it is clicked</p>
 21      *
 22      * @module mViewer
 23      * @namespace com.imaginea.mongov
 24      * @requires node
 25      * @constructor
 26      * @param node
 27      *		  <dd>(required) The node that is clicked</dd>
 28      * @param otherNodes
 29      *		  <dd>(required) Nodes which are there in the menu </dd>
 30      */
 31     MV.toggleClass = function (node, otherNodes) {
 32         otherNodes.removeClass('sel');
 33         node.addClass('sel');
 34     };
 35 }, '3.3.0', {
 36     requires: ["node"]
 37 });
