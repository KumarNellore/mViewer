215         function newObject() {

218         var getChildrenArray;
219         getChildrenArray = function(doc) {
220             var i, tempObject, key, value, newArray, childrenArray = [];
221             var allKeys = MV.getProperties(doc);
222             for (i = 0; i < allKeys.length; i++) {
223                 tempObject = newObject();
224                 key = allKeys[i];
225                 tempObject.key = key;
226                 value = doc[key];
227                 if (MV.isArray(value)) {
228                     newArray = [];
229                     tempObject.type = "BasicDBList";
230                     newArray = getChildrenArray(value);
231                     tempObject.kiddies = newArray;
232                 } else if (MV.isObject(value)) {
233                     newArray = [];
234                     tempObject.type = "BasicDBObject";
235                     newArray = getChildrenArray(value);
236                     tempObject.kiddies = newArray;
237                 } else {
238                     tempObject.value = value;
239                     if (key === "_id") {
240                         tempObject.type = "ObjectID";
241                     }
242                     else {
243                         tempObject.type = typeof value;
244                     }
245                 }
246                 childrenArray.push(tempObject);
247             }
248             return childrenArray;
249         };
