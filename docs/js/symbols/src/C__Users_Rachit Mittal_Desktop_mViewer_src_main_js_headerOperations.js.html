 16 YUI({
 17     filter: 'raw'
 18 }).use("alert-dialog", "utility", "node", "event-delegate", "stylize", "custom-datatable", function (Y) {
 19     Y.namespace('com.imaginea.mongoV');
 20     var MV = Y.com.imaginea.mongoV;
 21     var showServerStats = function () {
 22             var data = new YAHOO.util.XHRDataSource(MV.URLMap.serverStatistics(), {
 23                 responseType: YAHOO.util.XHRDataSource.TYPE_JSON,
 24                 responseSchema: {
 25                     resultsList: "response.result",
 26                     metaFields: {
 27                         startIndex: 'first_index',
 28                         recordsReturned: 'records_returned',
 29                         totalRecords: 'total_records'
 30                     }
 31                 }
 32             });
 33             data.sendRequest("", {
 34                 success: function (request, responseObject) {
 35                     MV.mainBody.set("innerHTML", '<div id="table"></div><div id="table-pagination"></div>');
 36                     var treebleData = MV.getTreebleDataForServerStats(responseObject);
 37                     treeble = MV.getTreeble(treebleData);
 38                     treeble.load();
 39                 },
 40                 failure: function (request, responseObject) {
 41                     MV.showAlertDialog("Failed: ServerStats could not be loaded! Please check if the app server is still running.", MV.warnIcon);
 42                     Y.log("Server Statistics could not be loaded. Response Status: [0]".format(responseObject.statusText), "error");
 43                 },
 44             });
 45         };
 46     var executeHeaderOption = function (eventObject) {
 47             menuOpt = eventObject.currentTarget.get("id");
 48             MV.toggleClass(eventObject.currentTarget, Y.all(".nav-cont li a"));
 49             // Clearing the content of main Body
 50             MV.mainBody.set("innerHTML", "");
 51             Y.one('#queryForm').set("innerHTML", "");
 52             if (menuOpt === "home") {
 53                 window.location = "home.html?tokenID=" + Y.one("#tokenID").get("value") + "&username=" + Y.one("#username").get("value") + "&host=" + Y.one("#host").get("value");
 54             } else if (menuOpt === "serverStats") {
 55                 showServerStats();
 56                 MV.header.set("innerHTML", "Server Statistics");
 57             } else if (menuOpt === 'graphs') {
 58                 MV.header.set("innerHTML", "");
 59                 window.open(MV.URLMap.graphs(), '_newtab');
 60             } else if (menuOpt === 'help') {
 61                 MV.header.set("innerHTML", "");
 62                 window.open(MV.URLMap.help(), '_newtab');
 63             } else if (menuOpt === 'troubleshoot') {
 64                 MV.header.set("innerHTML", "");
 65                 window.open(MV.URLMap.troubleShootPage(), '_newtab');
 66             }
 67         };
 68     var logout = function (eventObject) {
 69             var request = Y.io(MV.URLMap.logout(), {
 70                 method: "GET",
 71                 on: {
 72                     success: function (ioId, responseObject) {
 73                         parsedResponse = Y.JSON.parse(responseObject.responseText);
 74                         var response = parsedResponse.response.result;
 75                         if (response !== undefined) {
 76                             Y.log("Successfully logging out.", "info");
 77                             window.location = "loggedOut.html";
 78                         } else {
 79                             var error = parsedResponse.response.error;
 80                             MV.showAlertDialog("Cannot logout! [0]".format(MV.errorCodeMap[error.code]), MV.warnIcon);
 81                             Y.log("Could not logout. Message: [0], Code: [1]".format(error.message, error.code), "error");
 82                         }
 83                     },
 84                     failure: function (ioId, responseObject) {
 85                         MV.showAlertDialog("Could not send request. Check if app server is running. Response message: [0]".format(responseObject.statusText), MV.warnIcon);
 86                         Y.log("Could not send request to logout. Status text: [0] ".format(responseObject.statusText), "error");
 87                     }
 88                 }
 89             });
 90         };
 91     Y.delegate("click", executeHeaderOption, ".nav-cont", "li a");
 92     Y.on("click", logout, "#logout");
 93 });
