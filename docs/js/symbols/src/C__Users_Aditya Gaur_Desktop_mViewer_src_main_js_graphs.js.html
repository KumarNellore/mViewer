 16 YUI({
 17     filter: 'raw'
 18 }).use("io-base", "node", "json-parse", "utility", function (Y) {
 19     Y.namespace('com.imaginea.mongoV');
 20     var MV = Y.com.imaginea.mongoV;
 21     var pollingTime = 5000;
 22     var combinedChart;
 23     var chartOptions = "<table width='30%'>";
 24     chartOptions += "<tr>";
 25     chartOptions += "<td><button id='animation' class='btn'>Stop Animation</button></td>";
 26     chartOptions += "</tr>";
 27     chartOptions += "</table>";
 28     var drawChart = function () {
 29             YAHOO.widget.Chart.SWFURL = "lib/yui2/build/charts/assets/charts.swf";
 30             //Create a TabView
 31             var tabView = new YAHOO.widget.TabView();
 32             //Add a tab for the Query
 33             tabView.addTab(new YAHOO.widget.Tab({
 34                 label: 'Combined View',
 35                 content: '<span class="chart_title">Combined View</span> <div class="chart" id="combinedChart"></div><center><span class="chart_title">Time Stamp</span></center>' + chartOptions,
 36                 active: true,
 37                 width: '100%'
 38             }));
 39             tabView.addTab(new YAHOO.widget.Tab({
 40                 label: 'Queries',
 41                 content: '<span class="chart_title">Queries/Second</span><div class="chart" id="queryChart"></div><center><span class="chart_title">Time Stamp</span></center>',
 42                 width: '100%'
 43             }));
 44             //    Add a tab for the Update
 45             tabView.addTab(new YAHOO.widget.Tab({
 46                 label: 'Updates',
 47                 content: '<span class="chart_title">Updates/Second</span><div class="chart" id="updateChart"></div><center><span class="chart_title">Time Stamp</span></center>',
 48                 width: '100%'
 49             }));
 50             //    Add a tab for the Delete
 51             tabView.addTab(new YAHOO.widget.Tab({
 52                 label: 'Deletes',
 53                 content: '<span class="chart_title">Deletes/Second</span><div class="chart" id="deleteChart"></div><center><span class="chart_title">Time Stamp</span></center>',
 54                 width: '100%'
 55             }));
 56             //    Add a tab for the Insert
 57             tabView.addTab(new YAHOO.widget.Tab({
 58                 label: 'Inserts',
 59                 content: '<span class="chart_title">Inserts/Second</span><div class="chart" id="insertChart"></div><center><span class="chart_title">Time Stamp</span></center>',
 60                 width: '100%'
 61             }));
 62             //Append TabView to its container div
 63             tabView.appendTo('tabContainer');
 64             //--- data
 65             var queryData = new YAHOO.util.DataSource(MV.URLMap.graphQuery());
 66             queryData.connMethodPost = false;
 67             queryData.responseType = YAHOO.util.DataSource.TYPE_JSON;
 68             queryData.responseSchema = {
 69                 resultsList: "response.result",
 70                 fields: ["TimeStamp", "QueryValue", "UpdateValue", "DeleteValue", "InsertValue"]
 71             };
 72             var seriesDef = [{
 73                 displayName: "Query",
 74                 yField: "QueryValue",
 75                 style: {
 76                     color: 0xFF4848
 77                 }
 78             }, {
 79                 displayName: "Update",
 80                 yField: "UpdateValue",
 81                 style: {
 82                     color: 0xAE70ED
 83                 }
 84             }, {
 85                 displayName: "Delete",
 86                 yField: "DeleteValue",
 87                 style: {
 88                     color: 0x62D0FF
 89                 }
 90             }, {
 91                 displayName: "Insert",
 92                 yField: "InsertValue",
 93                 style: {
 94                     color: 0x2DC800
 95                 }
 96             }];
 97             combinedChart = new YAHOO.widget.LineChart("combinedChart", queryData, {
 98                 xField: "TimeStamp",
 99                 series: seriesDef,
100                 style: {
101                     legend: {
102                         display: "right",
103                         padding: 10,
104                         spacing: 5,
105                         font: {
106                             family: "Arial",
107                             size: 13
108                         }
109                     }
110                 }
111             });
112             var queryChart = new YAHOO.widget.LineChart("queryChart", queryData, {
113                 xField: "TimeStamp",
114                 yField: "QueryValue"
115             });
116             var updateChart = new YAHOO.widget.LineChart("updateChart", queryData, {
117                 xField: "TimeStamp",
118                 yField: "UpdateValue"
119             });
120             var deleteChart = new YAHOO.widget.LineChart("deleteChart", queryData, {
121                 xField: "TimeStamp",
122                 yField: "DeleteValue"
123             });
124             var insertChart = new YAHOO.widget.LineChart("insertChart", queryData, {
125                 xField: "TimeStamp",
126                 yField: "InsertValue"
127             });
128             var myCallback = {
129                 success: function () {
130                     queryChart._loadDataHandler.apply(queryChart, arguments);
131                     updateChart._loadDataHandler.apply(updateChart, arguments);
132                     deleteChart._loadDataHandler.apply(deleteChart, arguments);
133                     insertChart._loadDataHandler.apply(insertChart, arguments);
134                     combinedChart._loadDataHandler.apply(combinedChart, arguments);
135                 },
136                 failure: function () {
137                     alert("Polling failure for getting Graph info");
138                     Y.log("Polling failure for getting Graph info", "error");
139                 }
140             };
141             queryData.setInterval(pollingTime, null, myCallback);
142             //            Y.on("click", toggleChartVisibility, ["#checkButton1", "#checkButton2", "#checkButton3", "#checkbutton4"]);
143         };
144     //A function handler to use for successful requests:
145     var handleSuccess = function (ioId, responseObject) {
146             var parsedResponse = Y.JSON.parse(responseObject.responseText);
147             response = parsedResponse.response.result;
148             if (response !== undefined) {
149                 Y.log("Chart initiated");
150                 drawChart();
151             } else {
152                 MV.showAlertDialog("Error: [0]".format(parsedResponse.response.error.message), MV.warnIcon);
153                 Y.log("Error: [0]".format(parsedResponse.response.error.message), "error");
154             }
155         };
156     //A function handler to use for failed requests:
157     var handleFailure = function (ioId, o) {
158             MV.showAlertDialog("Could not send request!", MV.warnIcon);
159             Y.log("Sending request to initiate the graph failed. Response Status: [0]".format(responseObject.statusText), "error", "chart");
160         };
161     var cfg = {
162         method: "GET",
163         data: "pollingTime=" + pollingTime / 1000
164     };
165     var makeRequest = function () {
166             var fullUrl = window.location.search;
167             Y.one("#tokenID").set("value", fullUrl.substring(fullUrl.indexOf("=") + 1));
168             var sUrl = MV.URLMap.graphInitiate();
169             var request = Y.io(sUrl, cfg);
170         };
171     var stopAnimation = function (e) {
172             if (e.currentTarget.get("innerHTML") === 'Stop Animation') {
173                 combinedChart.setStyle("animationEnabled", false);
174                 e.currentTarget.set("innerHTML", "Start Animation");
175             } else {
176                 combinedChart.setStyle("animationEnabled", true);
177                 e.currentTarget.set("innerHTML", "Stop Animation");
178             }
179         };
180     //Subscribe our handlers to IO's global custom events:
181     Y.on('io:success', handleSuccess);
182     Y.on('io:failure', handleFailure);
183     Y.on('click', stopAnimation, '#animation');
184     // Make a request when the page loads
185     Y.on("load", makeRequest);
186 });
