  5     filter: 'raw'
  6 }).use("anim", "io-base", "json-parse", "node", "utility", function (Y) {
  7     var MV = Y.com.imaginea.mongoV;
  8     var tryLogin = function (e) {
  9             var username = Y.one("#username").get("value").trim();
 10             var password = Y.one("#password").get("value").trim();
 11             var port = Y.one("#port").get("value").trim();
 12             var host = Y.one("#host").get("value").trim();
 13             var resetBGColor = function () {
 14                     Y.all("input").setStyle("background", "#FFFFFF");
 15                 };
 16             var errorHandlerMap = {
 17                 "HOST_UNKNOWN": function () {
 18                     resetBGColor();
 19                     Y.one("#port").setStyle("background", "#FFEBE8");
 20                     Y.one("#host").setStyle("background", "#FFEBE8");
 21                 },
 22                 "MISSING_LOGIN_FIELDS": function () {
 23                     resetBGColor();
 24                     var fields = Y.all("input");
 25                     fields.each(function () {
 26                         if (this.get("value") === "") {
 27                             this.setStyle("backgroundColor", "#FFEBE8");
 28                         }
 29                     });
 30                 },
 31                 "ERROR_PARSING_PORT": function () {
 32                     resetBGColor();
 33                     Y.one("#port").setStyle("background", "#FFEBE8");
 34                 },
 35                 "PORT_OUT_OF_RANGE": function () {
 36                     resetBGColor();
 37                     Y.one("#port").setStyle("background", "#FFEBE8");
 38                 },
 39                 "INVALID_USERNAME": function () {
 40                     resetBGColor();
 41                     Y.one("#username").setStyle("background", "#FFEBE8");
 42                     Y.one("#password").setStyle("background", "#FFEBE8");
 43                 }
 44             };
 45             var request = Y.io(MV.URLMap.login(), {
 46                 data: "username=" + username + "&password=" + password + "&port=" + port + "&host=" + host,
 47                 method: "POST",
 48                 on: {
 49                     success: function (ioId, responseObject) {
 50                         parsedResponse = Y.JSON.parse(responseObject.responseText);
 51                         var response = parsedResponse.response.result;
 52                         if (response !== undefined) {
 53                             Y.log("Successfully logging in. TokenId: [0]".format(response.id), "info");
 54                             window.location = "home.html?tokenID=" + response.id + "&username=" + response.username + "&host=" + response.host;
 55                         } else {
 56                             var error = parsedResponse.response.error;
 57                             var errorDiv = Y.one("#errorMsg");
 58                             errorDiv.setStyle("display", "inline");
 59                             errorHandlerMap[error.code]();
 60                             errorDiv.set("innerHTML", MV.errorCodeMap[error.code] || "Error!");
 61                             Y.log("Could not login. Message: [0]".format(error.message), "error");
 62                         }
 63                     },
 64                     failure: function (ioId, responseObject) {
 65                         alert("Could not send request! Please check if application server is running.");
 66                         Y.log("Could not send request.", "error");
 67                     }
 68                 }
 69             });
 70         };
 71     var checkIfSubmitted = function (eventObject) {
 72             if (eventObject.keyCode === 13) {
 73                 tryLogin();
 74             }
 75         };
 76     Y.all("input").on("keyup", checkIfSubmitted);
 77     Y.one("#login").on('click', tryLogin);
