 26 YUI.add('custom-datatable', function(Y) {
 27     Y.namespace('com.imaginea.mongoV');
 28     /**
 29      * <p>custom-datatable creates a datatable with the following columns
 30      * <p>Currently it is used to make the datatable to show the collection and the
 31      * database statistics. </p>
 32      *
 33      * @module custom-datatable
 34      * @namespace com.imaginea.mongov
 35      * @requires "io-base", "node", "json-parse", "datatable-scroll", "datasource-io", "datasource-jsonschema", "datatable-datasource", "event-delegate"
 36      * @param path
 37      *          <dd>(required) This is the url to which the request will be sent to get the data</dd>
 38      * @param name
 39      *          <dd>(required)The name of the collection/database whose data is required.</dd>
 40      */
 41     var MV = Y.com.imaginea.mongoV;
 42     MV.createDatatable = function(path, name) {
 43         MV.mainBody.set("innerHTML", "");
 44         /**
 45          * <p>The function is used to make the column configuration for the datatable  
 46          * @param colKey
 47          * <dd>(required)The column name</dd>
 48          **/
 49         var createColumn = function(colKey) {
 50             return {
 51                 key: colKey,
 52                 sortable: true,
 53                 width: "210px"
 54             };
 55         };
 56         var cols = [createColumn("Key"), createColumn("Value"), createColumn("Type")];
 57         var ds = new Y.DataSource.IO({
 58             source: path
 59         });
 60         ds.plug(Y.Plugin.DataSourceJSONSchema, {
 61             schema: {
 62                 resultListLocator: "Result",
 63                 resultFields: ["Key", "Value", "Type"]
 64             }
 65         });
 66         var dt = new Y.DataTable.Base({
 67             columnset: cols,
 68             width: "697px"
 69         }).plug(Y.Plugin.DataTableDataSource, {
 70             datasource: ds,
 71             initialRequest: ""
 72         });
 73 
 74         ds.after("response", function() {
 75 
 76             MV.header.addClass('tab-cont');
 77             MV.header.set("innerHTML", "Statistics: " + name);
 78             dt.render("#" + MV.mainBody.get('id'));
 79         });
 80     };
 81 }, '3.3.0', {
 82     requires: ["io-base", "node", "json-parse", "datatable-scroll", "datasource-io", "datasource-jsonschema", "datatable-datasource", "event-delegate"]
 83 });
