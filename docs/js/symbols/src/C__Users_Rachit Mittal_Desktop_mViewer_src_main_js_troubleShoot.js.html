 17 YUI({
 18     filter: 'raw'
 19 }).use("io", "json", "node", "utility", function (Y) {
 20     var MV = Y.com.imaginea.mongoV;
 21     var trySetLevel = function (e) {
 22             var index = document.getElementById("logLevel").selectedIndex;
 23             var logValue = document.getElementById("logLevel").options[index].text;
 24             var request = Y.io(MV.URLMap.troubleShoot(), {
 25                 data: "level=" + logValue,
 26                 method: "GET",
 27                 on: {
 28                     success: function (ioId, responseObject) {
 29                         parsedResponse = Y.JSON.parse(responseObject.responseText);
 30                         var response = parsedResponse.response.result;
 31                         if (response !== undefined) {
 32                             Y.log("Successfully set the logging level", "info");
 33                             Y.one("#logMessage").set("innerHTML", response + "<br> Please send us mViewer.log stored in the logs directory of the application server in case of any problems.<br> agaur111@gmail.com , mtlrachit@gmail.com");
 34                         } else {
 35                             var error = parsedResponse.response.error;
 36                             var errorDiv = Y.one("#logMessage");
 37                             errorDiv.set("innerHTML", MV.errorCodeMap[error.code] || "Error!");
 38                             Y.log("Could not set logging level. Message: [0]".format(error.message), "error");
 39                         }
 40                     },
 41                     failure: function (ioId, responseObject) {
 42                         alert("Could not send request! Please check if application server is running.");
 43                         Y.log("Could not send request.", "error");
 44                     }
 45                 }
 46             });
 47         };
 48     Y.one("#logLevelChangeBtn").on('click', trySetLevel);
 49 });
