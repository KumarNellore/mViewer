 25 YUI({
 26     filter: 'raw'
 27 }).use("alert-dialog", "utility", "dialog-box", "yes-no-dialog", "io-base", "node", "json-parse", "event-delegate", "node-event-simulate", "stylize", "custom-datatable", function (Y) {
 28     // TODO: make loading panel module
 29     var dbDiv, loadingPanel;
 30     Y.namespace('com.imaginea.mongoV');
 31     var MV = Y.com.imaginea.mongoV; /* HANDLER FUNCTIONS */
 32     var parseAddCollResponse = function (responseObject) {
 33             var parsedResponse = Y.JSON.parse(responseObject.responseText);
 34             response = parsedResponse.response.result;
 35             if (response !== undefined) {
 36                 MV.showAlertDialog("[0] added to [1]".format(Y.one("#newName").get("value"), Y.one("#currentDB").get("value")), MV.infoIcon);
 37                 Y.log("[0] created in [1]".format(Y.one("#newName").get("value"), Y.one("#currentDB").get("value")), "info");
 38                 Y.one("#currentColl").set("value", "");
 39                 Y.one("#" + Y.one("#currentDB").get("value")).simulate("click");
 40             } else {
 41                 var error = parsedResponse.response.error;
 42                 MV.showAlertDialog("Could not add Collection! [0]".format(MV.errorCodeMap[error.code]), MV.warnIcon);
 43                 Y.log("Could not add Collection! [0]".format(MV.errorCodeMap[error.code]), "error");
 44             }
 45         };
 46     var showError = function (responseObject) {
 47             MV.showAlertDialog("Collection creation failed! Please check if app server is runnning.", MV.warnIcon);
 48             Y.log("Collection creation failed. Response Status: [0]".format(responseObject.statusText), "error");
 49         };
 50     var sendDropDBRequest = function () {
 51             Y.log("Preparing to send request to drop DB", "info");
 52             var request = Y.io(MV.URLMap.dropDB(), {
 53                 method: "POST",
 54                 on: {
 55                     success: function (ioId, responseObject) {
 56                         var parsedResponse = Y.JSON.parse(responseObject.responseText);
 57                         if (parsedResponse.response.result !== undefined) {
 58                             MV.showAlertDialog("[0] is dropped! ".format(Y.one("#currentDB").get("value")), MV.infoIcon, function () {
 59                                 window.location = "home.html?tokenID=" + Y.one("#tokenID").get("value") + "&username=" + Y.one("#username").get("value") + "&host=" + Y.one("#host").get("value");
 60                             });
 61                             Y.log("[0] dropped".format(Y.one("#currentDB").get("value")), "info");
 62                             Y.one("#currentDB").set("value", "");
 63                         } else {
 64                             var error = parsedResponse.response.error;
 65                             MV.showAlertDialog("Could not drop: [0]. [1]".format(Y.one("#currentDB").get("value"), MV.errorCodeMap[error.code]), MV.warnIcon);
 66                             Y.log("Could not drop: [0], Response Recieved: [1], ErrorCode: [2]".format(Y.one("#currentDB").get("value"), error.message, error.code), "error");
 67                         }
 68                     },
 69                     failure: function (ioId, responseObject) {
 70                         Y.log("Could not drop: [0]. Status Text: [1]".format(Y.one("#currentDB").get("value"), responseObject.statusText), "error");
 71                         MV.showAlertDialog("Could not drop: [0], Status Text: [2]".format(Y.one("#currentDB").get("value"), responseObject.statusText), MV.warnIcon);
 72                     }
 73                 }
 74             });
 75             this.hide();
 76         };
 77     var handleNo = function (dialog) {
 78             this.hide();
 79         };
 80     var executeContextMenuOption = function (eventType, args) {
 81             var menuItem = args[1]; // The MenuItem that was clicked
 82             Y.one("#currentDB").set("value", this.contextEventTarget.id);
 83             MV.toggleClass(Y.one("#" + Y.one("#currentDB").get("value")), Y.all("#dbNames li"));
 84             switch (menuItem.index) {
 85             case 0:
 86                 // Delete database
 87                 dialog = MV.showYesNoDialog("Do you really want to drop the Database?", sendDropDBRequest, handleNo);
 88                 break;
 89             case 1:
 90                 // add collection
 91                 var form = "addColDialog";
 92                 MV.getDialog(form, parseAddCollResponse, showError);
 93                 break;
 94             case 2:
 95                 // show statistics
 96                 MV.hideQueryForm();
 97                 MV.createDatatable(MV.URLMap.dbStatistics(), Y.one("#currentDB").get("value"));
 98                 break;
 99             }
100         };
101     var dbContextMenu = new YAHOO.widget.ContextMenu("dbContextMenuID", {
102         trigger: "dbNames",
103         itemData: ["Delete Database", "Add Collection", "Statistics"]
104     });
105     dbContextMenu.render("dbContextMenu");
106     dbContextMenu.clickEvent.subscribe(executeContextMenuOption);
107     // A function handler to use for successful requests to get DB names:
108 
109     function parseGetDBResponse(ioId, responseObject) {
110         Y.log("Response Recieved of get DB request", "info");
111         try {
112             var parsedResponse = Y.JSON.parse(responseObject.responseText);
113             if (parsedResponse.response.result !== undefined) {
114                 var info, index, dbNames = "";
115                 for (index = 0; index < parsedResponse.response.result.length; index++) {
116                     dbNames += "<li id='[0]' >[1]</li>".format(parsedResponse.response.result[index], parsedResponse.response.result[index]);
117                 }
118                 if (index === 0) {
119                     dbDiv.set("innerHTML", "No Databases");
120                 }
121                 dbDiv.set("innerHTML", dbNames);
122                 Y.one('#user').set("innerHTML", Y.one("#username").get("value"));
123                 Y.one('#hostname').set("innerHTML", Y.one("#host").get("value"));
124                 loadingPanel.hide();
125                 Y.log("Database Names succesfully loaded", "info");
126             } else {
127                 var error = parsedResponse.response.error;
128                 Y.log("Could not load databases. Message from server: [0]. Error Code from server:[1] ".format(error.message, error.code), "error");
129                 MV.showAlertDialog(MV.errorCodeMap[error.code], MV.warnIcon);
130                 loadingPanel.hide();
131             }
132         } catch (e) {
133             MV.showAlertDialog(e, MV.warnIcon);
134         }
135     }
136     // A function handler to use for failed requests to get DB names:
137 
138     function displayError(ioId, responseObject) {
139         Y.log("Could not load the databases", "error");
140         Y.log("Status code message: [0]".format(responseObject.statusText), "error");
141         loadingPanel.hide();
142         MV.showAlertDialog("Could not load collections! Please check if the app server is running. Status Text: [0]".format(responseObject.statustext), MV.warnIcon);
143     }
144     var getParameters = function () {
145             var params = [];
146             var fullUrl = window.location.search;
147             while (fullUrl.indexOf("&") !== -1) {
148                 params.push(fullUrl.substring(fullUrl.indexOf("=") + 1, fullUrl.indexOf("&")));
149                 fullUrl = fullUrl.substring(fullUrl.indexOf("&") + 1);
150             }
151             params.push(fullUrl.substring(fullUrl.indexOf("=") + 1));
152             return params;
153         };
154     var requestDBNames = function () {
155             loadingPanel = new LoadingPanel("Loading Databases...");
156             loadingPanel.show();
157             dbDiv = Y.one('#dbNames ul.lists');
158             var params = getParameters();
159             Y.log(params[0], "info");
160             Y.one("#tokenID").set("value", params[0]);
161             Y.one("#username").set("value", params[1]);
162             Y.one("#host").set("value", params[2]);
163             var request = Y.io(MV.URLMap.getDBs(),
164             // configuration for loading the database names
165             {
166                 method: "GET",
167                 on: {
168                     success: parseGetDBResponse,
169                     failure: displayError
170                 }
171             });
172             Y.log("Sending request to load DB names", "info");
173         }; /* EVENT LISTENERS */
174     // Make a request to load Database names when the page loads
175     Y.on("load", requestDBNames);
176 });
