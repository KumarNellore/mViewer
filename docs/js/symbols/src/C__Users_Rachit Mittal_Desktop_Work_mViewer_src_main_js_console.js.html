  5 YUI({
  6 		filter : 'raw'
  7 	}).use("console", "console-filters", "dd-plugin", function(Y) {
  8 		// creating a console screen
  9 		var globalConsole;
 10 		globalConsole = new Y.Console({
 11 			logSource : Y.Global,
 12 			strings : {
 13 				title : 'Console',
 14 				pause : 'Pause',
 15 				clear : 'Clear',
 16 				collapse : 'Collapse',
 17 				expand : 'Expand'
 18 			},
 19 			visible : false
 20 		}).plug(Y.Plugin.ConsoleFilters).plug(Y.Plugin.Drag, {
 21 			handles : ['.yui3-console-hd']
 22 		}).render();
 23 		// Set up the button listeners
 24 		function toggle(e, globalConsole) {
 25 			if (globalConsole.get('visible')) {
 26 				globalConsole.hide();
 27 				this.set('innerHTML', 'Show console');
 28 			} else {
 29 				globalConsole.show();
 30 				globalConsole.syncUI(); // to handle any UI changes queued while hidden.
 31 				this.set('innerHTML', 'Hide console');
 32 			}
 33 		}
 34 		Y.on('click', toggle, '#toggle_console', null, globalConsole);
 35 	});
