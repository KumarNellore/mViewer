 25 YUI.add('stylize', function (Y) {
 26     Y.namespace('com.imaginea.mongoV');
 27     var MV = Y.com.imaginea.mongoV;
 28     /**
 29      * <p>toggleClass adds a class to the menu option as soon as it is clicked</p>
 30      *
 31      * @module mViewer
 32      * @namespace com.imaginea.mongov
 33      * @requires node
 34      * @constructor
 35      * @param node
 36      *		  <dd>(required) The node that is clicked</dd>
 37      * @param otherNodes
 38      *		  <dd>(required) Nodes which are there in the menu </dd>
 39      */
 40     MV.toggleClass = function (node, otherNodes) {
 41         otherNodes.removeClass('sel');
 42         node.addClass('sel');
 43     };
 44 }, '3.3.0', {
 45     requires: ["node"]
 46 });
