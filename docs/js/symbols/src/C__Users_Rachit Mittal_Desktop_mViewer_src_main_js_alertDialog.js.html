 17 YUI.add('alert-dialog', function (Y) {
 18     Y.namespace('com.imaginea.mongoV');
 19     Y.com.imaginea.mongoV.showAlertDialog = function (msg, ico, handler) {
 20         var simpleDialog = new YAHOO.widget.SimpleDialog("simpledialog", {
 21             width: "300px",
 22             fixedcenter: true,
 23             visible: false,
 24             draggable: true,
 25             close: false,
 26             text: msg,
 27             icon: ico || YAHOO.widget.SimpleDialog.ICON_INFO,
 28             constraintoviewport: true,
 29             zIndex: 10,
 30             buttons: [{
 31                 text: "OK",
 32                 handler: handler ||
 33                 function () {
 34                     this.hide();
 35                 },
 36                 isDefault: true
 37             }]
 38         });
 39         simpleDialog.setHeader("Info");
 40         simpleDialog.render("alertDialogContainer");
 41         simpleDialog.show();
 42     };
 43 }, '3.3.0', {
 44     requires: []
 45 });
