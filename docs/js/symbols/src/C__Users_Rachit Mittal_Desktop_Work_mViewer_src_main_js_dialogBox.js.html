 25 YUI.add('dialog-box', function (Y) {
 26     Y.namespace('com.imaginea.mongoV');
 27     var MV = Y.com.imaginea.mongoV;
 28     MV.getDialog = function Dialog(form, successHandler, failureHandler) {
 29         YAHOO.util.Dom.removeClass(form, "yui-pe-content");
 30         var handleCancel = function () {
 31                 this.cancel();
 32             };
 33         var addColDialoghandleSubmit = function () {
 34                 Y.log("Submit handler for add collection called","info");
 35                 var newCollInfo = this.getData();
 36                 if (newCollInfo.name === "") {
 37                     MV.showAlertDialog("Please enter the name.");
 38                 } else {
 39                     Y.one("#newName").set("value", newCollInfo.name);
 40                     Y.one("#" + form + " .bd form").setAttribute("action", MV.URLMap.insertColl());
 41                     this.submit();
 42                 }
 43             };
 44         var addDBDialoghandleSubmit = function () {
 45                 var newDBInfo = this.getData();
 46                 if (newDBInfo.name === "") {
 47                     MV.showAlertDialog("Please enter the name.");
 48                 } else {
 49                     Y.one("#newName").set("value", newDBInfo.name);
 50                     Y.one("#" + form + " .bd form").setAttribute("action", MV.URLMap.insertDB());
 51                     this.submit();
 52                 }
 53             };
 54         var addDocDialoghandleSubmit = function () {
 55                 var newDoc = this.getData().document;
 56                 try {
 57                     Y.JSON.parse(newDoc);
 58                     Y.one("#" + form + " .bd form").setAttribute("action", MV.URLMap.insertDoc());
 59                     this.submit();
 60                 } catch (e) {
 61                     MV.showAlertDialog("Please enter the new document in JSON format", MV.warnIcon);
 62                     Y.log("New Document format not JSON", "error");
 63                 }
 64             };
 65         var sumbitHandlerMap = {
 66             "addColDialogSubmitHandler": addColDialoghandleSubmit,
 67             "addDBDialogSubmitHandler": addDBDialoghandleSubmit,
 68             "addDocDialogSubmitHandler": addDocDialoghandleSubmit
 69         };
 70         var dialogBox = new YAHOO.widget.Dialog(form, {
 71             width: "30em",
 72             fixedcenter: true,
 73             visible: false,
 74             effect: {
 75                 effect: YAHOO.widget.ContainerEffect.SLIDE,
 76                 duration: 0.25
 77             },
 78             constraintoviewport: true,
 79             buttons: [{
 80                 text: "Submit",
 81                 handler: sumbitHandlerMap[form + "SubmitHandler"] ||
 82                 function () {
 83                     this.submit();
 84                 },
 85                 isDefault: true
 86             }, {
 87                 text: "Cancel",
 88                 handler: handleCancel
 89             }]
 90         });
 91         dialogBox.callback = {
 92             success: successHandler,
 93             failure: failureHandler
 94         };
 95         dialogBox.render();
 96         dialogBox.show();
 97         return dialogBox;
 98     };
 99 }, '3.3.0', {
100     requires: ["utility", "node", "alert-dialog"]
101 });
